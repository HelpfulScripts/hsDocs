  12export function setRestartTimer(hour:number, minute:number, recipient:Recipient) {
  13    const now = new Date();
  14//    const restartTime = new Date(now.getTime()+10*60*1000); // in 10 min
  15
  16    const restartTime = new Date(now.getTime()+10*60*1000);
  17    restartTime.setHours(hour,minute,0,0);
  18    if (restartTime < now) { restartTime.setDate(restartTime.getDate()+1); }
  19    
  20    const diff = restartTime.getTime() - now.getTime();
  21    Promise.resolve().then(delay(diff)).then(Exec.restartFn);
  22    return OSA.iMessage('next restart in ' + Math.round(10*diff/1000/60/60)/10 + ' hours', [recipient]);
  23}
