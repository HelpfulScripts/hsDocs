  10export interface Command {
  11    command: string;
  12    commandFn: (query:string[]) => Promise;
  13    desc?: string;
  14}

  16export interface Query {
  17    event: any;
  18    command: string;

  22const gCommands = [];

  24function interpretCommand(query:Query) {
  25    const cmdObj = gCommands[query.command.split(' ')[0]];
  26    if (cmdObj) { 
  27        log.info('received command \'' + query.command + '\' for event '  + query.event);
  28        try { cmdObj.commandFn(query); }
  29        catch(err) { console.trace('error executing command ' + query.command, err); }
  30    } else if (query.event === 'script') {
  31        log.info('received script response \'' + query.command  + '\'');
  32    } else { 
  33        log.info('received unkwon command \'' + query.command  + '\' for event '  + query.event);
  34        Exec.sayFn(query);
  35    }
  36}

  38/**
  39 * opens an http server on the speicfied `port` and listens for requests. 
  40 * The `query` part of the request is passed into the callback for any registered command.   
  41 * `query.command` is used to identify the intended command.
  42 * @param number port the port on which to listen for requests.
  43 * @param function defaultCallback function to call for unkown commands.
  44 */
  45export const listen = (port:number) => {
  46    try { 
  47        http.createServer((request, response) => {
  48            const query:Query = url.parse(request.url, true).query;
  49            log.debug('received query ' + JSON.stringify(query));
  50            if (query.command) { query.command = query.command.toLowerCase().trim(); }
  51            else { 
  52                log.error('query.command is undefined for ' + request.url); 
  53                return; 
  54            }
  55
  56            interpretCommand(query);
  57
  58            request.setEncoding("utf8");
  59            request.addListener("data", function(/*postDataChunk*/) { });
  60            request.addListener("end", function() {
  61                response.writeHead(200, { "Content-Type": "text/plain" });
  62                response.write('');
  63                response.end();
  64            });
  65        }).listen(port);
  66    } 
  67    catch(e) { log.error(e); }
  68    log.info('server started on http://' + os.hostname() + ':' + port + '/');
  69};

  71/**
  72 * adds a command and a corresponding callback function to the list of 
  73 * registered commands.
  74 * @param string cmd the command to add.
  75 * @param function cmdFn the function to call when the command is received.
  76 * @param string desc optional; 
  77 */
  78export const addCommand = (cmd:string, cmdFn:any, desc?:string) => {
  79    log.debug('adding command ' + cmd);
  80    var obj = {command: cmd, commandFn: cmdFn, desc:desc||cmd};
  81    gCommands.push(obj);
  82    gCommands[cmd] = obj;
  83};
  84
  85/**
  86 * adds a command and a corresponding callback function to the list of 
  87 * registered commands.
  88 * @return string[ ] an array of command strings
  89 */
  90export const getCommands = () => {
  91    log.info('getting list of command');
  92    return  gCommands.map((c) => c.desc);
  93};
