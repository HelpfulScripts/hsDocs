  31    /**
  32     * promised to send `cmd` to the foscam camera specified in `options`
  33     * @param cmd the command string to send
  34     * @param {dynData an http options object
  35     */
  36    private sendCommandToFoscam(cmd:string, dynData:any={}):Promise {
  37        let binary = false;
  38        const settings = this.getSettings();
  39        let url = `http://${settings.host}:${settings.port}${this.path}`;
  40        let options:any = {};
  41        if (cmd.trim() === '') {    // retrieve url from `dynData`
  42            options.path = dynData.html.body.img.attrs.src;
  43            options.headers.referer = dynData.url;
  44            log.debug(`getting image from ${options.path}`);
  45            binary = true;
  46        } else {
  47            options.path = `${this.path}&cmd=${cmd}`; 
  48            log.info('sending ' + cmd + ' to ' + this.getName());
  49        }
  50        return http.get(url)
  51            .then(this.decodeXmlResult.bind(this))
  52            .then(res => {
  53                if (res) {
  54                    if (res.CGI_Result) {
  55                        res = res.CGI_Result;
  56                        if (res.result !== '0') { throw new Error(`Unsuccessful Command ${cmd}: ${res.result}`); }
  57                    }
  58                    res.url = url;
  59                    res.cmd = cmd;
  60                }
  61                log.debug(`result for '${cmd}': ${inspect(res)}`);
  62                return res;
  63            });
  64    }

  73        for (let i=0; i<10; i++) { area = area + '&area' + i + '=1023'; }
  74        armCmd = armCmd  +
  75            '&sensitivity=' + sensitivity +

  83    /**
  84     * captures a snapshot from the device and saves it to gSnapshotDir
  85     * @param string deviceName the name of the device
  86     */
  87    snapPicture():Promise {
  88        const options = this.getSettings();
  89        if (!options) {
  90            log.error('device ' + this.getName() + ' not available');
  91            return Promise.reject();
  92        } else {
  93            return this.sendCommandToFoscam(snapPicCmd)
  94            .then((data:any) => {
  95                return this.sendCommandToFoscam('', data);    // no command word triggers simple request for options.path
  96            })
  97            .then((picData:any) => {
  98                let fileName = this.getRecordingDir() + date("snapshot_%YYYY%MM%DD-%hh-%mm-%ss.jpg");
  99                log.info(`saving snapshot from ${this.getName()} (len=${picData.length}) at ${fileName}`);
 100                fs.writeStream(fileName, picData);
 101            })
 102            .catch(log.error);
 103        }
 104    }

 151     * promises to check whether the device is armed or not. 
 152     * @param string deviceName the name of the device
 153     * @param function cb the function to call with the result
 154     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed).

 156    armStatus():Promise {
 157        // results in {motionDetectAlarm: '0'- disarmed, '1'-no alarm, '2'- detect alarm}
 158        return this.sendCommandToFoscam(statusCmd)  
 159            // resolves to True (armed) of False (disarmed)
 160            .then((result:any) => this.armed = (result.motionDetectAlarm !== '0'))   
 161            .catch(err => {
 162                log.error(err);
 163                throw err;
 164            });

 167    /**
 168     * returns a promise to arm or disarm the device.
 169     * @param string deviceName the name of the device
 170     * @param boolean arm `true` will arm, `false` will disarm the device.
 171     * @return Promise a promise that resolves to True (status: armed) or False (status: disarmed). 
 172     */
 173    arm(arm:boolean):Promise {
 174        const cmd = arm? (armCmd+'&linkage=' + (linkage + (this.getAudible()?ring:0))) : disarmCmd;
 175        return this.sendCommandToFoscam(cmd)        
 176            .then((result) => this.armed = result)  // resolves to the arming status of the device (true or false)
 177            .catch(err => {
 178                log.error(err);
 179                throw err;
 180            });
 181    }
