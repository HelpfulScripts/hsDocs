  29function armingCall(deviceCalls: Promise[]):Promise {
  30    return Promise.race([
  31        Promise.all(deviceCalls),   // call all alarm devices 
  32        timeout(20000)              // and timeout after 20s
  33        ])
  34    .then(() => {})
  35    .catch(err => {
  36        log.error(err);
  37        const resp = 'some device status results missing';
  38        log.error(resp);
  39    });
  40}

  81/**
  82 * request to snap a picture. If a device name is specified, a snapshot from that device will be requested.
  83 * If no name is specified, a snapshot from each available camera willbe requested.
  84 * @param query : `snap [deviceName]`
  85 */
  86export const snapFn = (query:Query):Promise => {
  87    let name = query.command.split(' ')[1];
  88    if (!name) { 
  89        return Promise.all(DeviceList.getDevices().map(device =>
  90            device.hasVideo()? (device).snapPicture() : Promise.resolve(true)));
  91    } else {
  92        const device = DeviceList.getDevice(name);
  93        return device.hasVideo()? (device).snapPicture() : Promise.resolve(true);
  94    }
  95};

 108/**
 109 * arms all armable devices. If `away` is specified, it also sets the device's audible alarm repsonses
 110 * Format:
 111 * `alarm [away]` 
 112 * @param query 
 113 */
 114export const armFn = (query:Query):Promise => {
 115    const away = (query.command.split(' ')[1] === 'away');
 116    log.info('arming system' + (away?' for away mode':''));
 117    const alarmDevices = DeviceList.getDevices()
 118        .filter((dev:Device) => dev.hasAlarm());
 119    return armingCall(alarmDevices.map((dev:AlarmDevice) => 
 120            dev.setAudible(away).then(() => dev.arm(true))))
 121        .then(() => alarmDevices.map((dev:AlarmDevice) => dev.isArmed()));
 122};

 124export const disarmFn = (query:Query):Promise => {

 126        .filter((dev:Device) => dev.hasAlarm());
 127    return OSA.quit('FaceTime')
 128        .then(() => armingCall(alarmDevices.map((dev:AlarmDevice) => 
 129            dev.setAudible(false).then(() => dev.arm(false)))))
 130        .then(() => alarmDevices.map((dev:AlarmDevice) => dev.isArmed()));
 131};
