  39async function activitySteps(rtcFilter:boolean) {
  40    getBasicData();
  41    [SPs, COFIs] = await getSPsAndCOFIs();
  42    await getSPactivities(activityList);
  43    await getCOFIactivities(activityList, COFIs);
  44    // await getTOPactivities();
  45    fs.writeJsonFile(`./_data/${currentYear}/activities.json`, activityList);
  46    const current:Act[] = await getCurrentYearActivities(activityList, currentYear);
  47    const contr = <{string:contrType}>{};
  48    await Promise.all(current.map(async (a:Act, i:number, as:Act[]) => {
  49        return await Promise.all(a.yrs.map(async (yr:ActYear) => {
  50            const folder = `${a.name} ${a.id}`;
  51            fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/currentYear.json`, yr);
  52            let dc:DepContrType[] = await CRIS.get('activity', `activities/${a.id}/years/${yr.id}/department-contributions`);
  53            if (!dc.filter || !yr.competenceField) { dc = []; }
  54            if (rtcFilter) { dc = getRTCNAActivity(dc, yr.competenceField.name); }       
  55            if(dc.length>0) { 
  56                log.debug(`${a.name}...dc: ${dc.length}`);    
  57                fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/contribution.json`, dc);
  58                return await getActivity(a, yr, folder, dc, contr);
  59            }
  60            return dc;
  61        }))
  62        .catch(err => log.error(`error in ${a.name} ${a.id}: ${err}\n${err.stack}\n${log.inspect(a,1)}`));
  63        }
  64    ));
  65    getRTCNAcontributions(contr, currentYear);
  66    // getRTCNAfunding(contr, currentYear);
  67    log.info(`found ${Object.keys(contr).length} contributions`);
  68    fs.writeJsonFile(`./_data/${currentYear}/contributions.json`, contr);
  69}

  71export function activities(yr:number, online:boolean) {
  72    const rtcFilter = false;
  73    currentYear = yr;
  74    CRIS.setOnline(online);
  75    log.info(`getting ${currentYear} activities`);
  76    activitySteps(rtcFilter).catch(log.error);
  77}

  87async function getActivity(a: Act, yr:ActYear, folder:string, dc:DepContrType[], contr:{string:contrType}) {
  88    CRIS.get('activity', `activities/${a.id}/years/${yr.id}/attachments`);
  89    const [vba, commitments, funding, pfp, pc, mc, customers] = await Promise.all([
  90        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/value-benefit-analysis`,false),
  91        CRIS.get('activity', `activities/${a.id}/fundings/commitments`),
  92        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/financials/fundings`),
  93        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/financials/public-fundings`),
  94        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/costplan/capacity-costs`),
  95        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/costplan/material-costs`),
  96        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/customers`),
  97    ]).catch(log.error); 
  98    log.debug(`${a.name}...details loaded`);    
  99    contr[a.name] = contr[a.name] || []; 
 100    contr[a.name].push(...dc.map((c:any):contrType => { return {
 101        activity:    a,             year:         yr,   vba: vba,
 102        commitments: commitments,   contribution: c,
 103        persCosts:   pc,            matCosts:     mc,
 104        customers:   customers,     funding:      funding,
 105        pfp:         pfp,           cofi:         yr.competenceField.name
 106    };}));
 107    log.debug(`${a.name}...found ${contr[a.name].length} contributions`);    
 108    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/vba.json`, vba); 
 109    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/commitments.json`, commitments); 
 110    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/fundings.json`, funding); 
 111    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/publicfundings.json`, pfp); 
 112    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/pc.json`, pc);
 113    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/mc.json`, mc); 
 114    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/customers.json`, customers); 
 115    return [commitments, funding, pfp, pc, mc, customers];
 116}

 120function getRTCNAcontributions(rtcContributions:{string:contrType}, yr:number) {
 121    const table:any[] = [];
 122    table.push(['Prj','CoFi','Title','ID','Resp','Years',
 123        'Start','End','Lifetime Costs',  
 124        `${yr} PY`,`${yr} PC`,`${yr} MC`,`${yr} Depts`, 'CW3 PY', 'Dept','PY','Hours','PC','MC',
 125        'Cost Share', 'Total Funding Share', 'Pfp Share', 'Direct Share', 'Credit Share', 'Source', 'Year', 
 126        'Type', 'Duration'
 127    ]);
 128    log.info(`contributions...${Object.keys(rtcContributions).length}`);
 129    Object.keys(rtcContributions).map(prj =>  
 130        rtcContributions[prj]
 131        .filter((p:contrType) => p.year.year === yr)
 132        .filter((p:contrType) => p.contribution.personalCost+p.contribution.materialCost>0 )
 133        .map((p:contrType) => {
 134            const cyf = funding(p, yr);
 135            const costs = totalCosts(p, yr);
 136            const CW3 = p.contribution.personalCapacity<=0? 0 : 
 137                costs.depts
 138                    .filter(d=> d[0].startsWith('CW3'))
 139                    .reduce((p:number, c:[string,number]) => p+c[1], 0);
 140            let costShare = 0;
 141            if(costs.pc + costs.mc>0) { costShare = (p.contribution.personalCost + p.contribution.materialCost) / (costs.pc + costs.mc); }
 142            if (CW3>0 && p.contribution.deptName==="ARE") { costShare *= CW3/p.contribution.personalCapacity; }
 143            const title = p.year.currentVersion.titleEnglish || p.year.currentVersion.titleGerman;
 144            const row = [
 145                /* 'Prj'            */ p.activity.name,
 146                /* 'CoFi'           */ p.cofi,
 147                /* 'Title'          */ title? title.trim().replace(/[,|\r|\n]/g,';') : '???',
 148                /* 'ID'             */ p.contribution.id,
 149                /* 'Resp'           */ p.year.responsible.name,
 150                /* 'Years'          */ '>' + Object.keys(costs.lifeTimeCosts).join(' '),
 151                /* 'Start'          */ p.year.currentVersion.start,
 152                /* 'End'            */ p.year.currentVersion.end,
 153                /* 'Lifetime Costs' */ Object.keys(costs.lifeTimeCosts).reduce((acc, yr)=> acc+costs.lifeTimeCosts[yr], 0),
 154                /* '2018 PY'        */ costs.py,
 155                /* '2018 PC'        */ costs.pc,
 156                /* '2018 MC'        */ costs.mc,
 157                /* '2018 Depts'     */ costs.depts
 158                                        .sort((a:[string, number],b:[string, number])=>b[1]-a[1])
 159                                        .map((d:[string, number])=> `${d[0]}(${d[1]})`)
 160                                        .join(' '),
 161                /* 'CW3 PY'          */ CW3,
 162                /* 'Dept.'          */ p.contribution.deptName,
 163                /* 'PY'             */ p.contribution.personalCapacity,
 164                /* 'Hours'          */ p.contribution.personalHours,
 165                /* 'PC'             */ p.contribution.personalCost,
 166                /* 'MC'             */ p.contribution.materialCost,
 167                /* 'Cost Share'     */ costShare,
 168                /* 'Total Funding'  */ cyf.totalFunding*costShare, 
 169                /* 'Pfp'            */ cyf.publicFunding*costShare, 
 170                /* 'Direct'         */ cyf.direct*costShare, 
 171                /* 'Credit'.        */ cyf.credit*costShare,
 172                /* 'Source'         */ cyf.source.join(' '),
 173                /* 'Year'           */ cyf.year,
 174                /* 'Type'           */ p.activity.type,
 175                /* 'Duration'       */ ms.toDays(Date.parse(p.year.currentVersion.end) - Date.parse(p.year.currentVersion.start))/365,
 176            ];
 177            table.push(row);
 178        })
 179    );
 180    log.info(`writing departmentContributions.csv, ${table.length} rows`);
 181    fs.writeTextFile(`./_data/${yr}/departmentContributions.csv`, table.join('\n'));
 182}
