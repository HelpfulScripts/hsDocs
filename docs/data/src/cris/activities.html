  38async function activitySteps(rtcFilter:boolean) {
  39    getBasicData();
  40    [SPs, COFIs] = await getSPsAndCOFIs();
  41    await getSPactivities(activityList);
  42    await getCOFIactivities(activityList, COFIs);
  43    // await getTOPactivities();
  44    fs.writeJsonFile(`./_data/${currentYear}/activities.json`, activityList);
  45    const current:Act[] = await getCurrentYearActivities(activityList, currentYear);
  46    const contr = <{string:contrType}>{};
  47    await Promise.all(current.map(async (a:Act, i:number, as:Act[]) => {
  48        return await Promise.all(a.yrs.map(async (yr:ActYear) => {
  49            const folder = `${a.name} ${a.id}`;
  50            fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/currentYear.json`, yr);
  51            let dc:DepContrType[] = await CRIS.get('activity', `activities/${a.id}/years/${yr.id}/department-contributions`);
  52            if (!dc.filter || !yr.competenceField) { dc = []; }
  53            if (rtcFilter) { dc = getRTCNAActivity(dc, yr.competenceField.name); }       
  54            if(dc.length>0) { 
  55                log.debug(`${a.name}...dc: ${dc.length}`);    
  56                fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/contribution.json`, dc);
  57                return await getActivity(a, yr, folder, dc, contr);
  58            }
  59            return dc;
  60        }))
  61        .catch(err => log.error(`error in ${a.name} ${a.id}: ${err}\n${err.stack}\n${log.inspect(a,1)}`));
  62        }
  63    ));
  64    getRTCNAcontributions(contr, currentYear);
  65    // getRTCNAfunding(contr, currentYear);
  66    log.info(`found ${Object.keys(contr).length} contributions`);
  67    fs.writeJsonFile(`./_data/${currentYear}/contributions.json`, contr);
  68}

  70export function activities(yr:number) {
  71    const rtcFilter = false;
  72    currentYear = yr;
  73    log.info(`getting ${currentYear} activities`);
  74    activitySteps(rtcFilter).catch(log.error);
  75}

  85async function getActivity(a: Act, yr:ActYear, folder:string, dc:DepContrType[], contr:{string:contrType}) {
  86    CRIS.get('activity', `activities/${a.id}/years/${yr.id}/attachments`);
  87    const [vba, commitments, funding, pfp, pc, mc, customers] = await Promise.all([
  88        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/value-benefit-analysis`),
  89        CRIS.get('activity', `activities/${a.id}/fundings/commitments`),
  90        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/financials/fundings`),
  91        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/financials/public-fundings`),
  92        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/costplan/capacity-costs`),
  93        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/costplan/material-costs`),
  94        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/customers`),
  95    ]).catch(log.error); 
  96    log.debug(`${a.name}...details loaded`);    
  97    contr[a.name] = contr[a.name] || []; 
  98    contr[a.name].push(...dc.map((c:any):contrType => { return {
  99        activity:    a,             year:         yr,   vba: vba,
 100        commitments: commitments,   contribution: c,
 101        persCosts:   pc,            matCosts:     mc,
 102        customers:   customers,     funding:      funding,
 103        pfp:         pfp,           cofi:         yr.competenceField.name
 104    };}));
 105    log.debug(`${a.name}...found ${contr[a.name].length} contributions`);    
 106    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/vba.json`, vba); 
 107    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/commitments.json`, commitments); 
 108    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/fundings.json`, funding); 
 109    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/publicfundings.json`, pfp); 
 110    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/pc.json`, pc);
 111    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/mc.json`, mc); 
 112    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/customers.json`, customers); 
 113    return [commitments, funding, pfp, pc, mc, customers];
 114}
