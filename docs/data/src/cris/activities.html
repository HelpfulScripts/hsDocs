  28async function activitySteps() {
  29    getBasicData();
  30    [SPs, COFIs] = await getSPsAndCOFIs();
  31    await getSPactivities(activityList);
  32    await getCOFIactivities(activityList, COFIs);
  33    // await getTOPactivities();
  34    await fs.writeJsonFile(`./_data/${currentYear}/activities.json`, activityList);
  35    const current:Act[] = await getCurrentYearActivities(activityList, currentYear);
  36    const contr = {};
  37    await Promise.all(current.map(async (a:Act, i:number, as:any[]) => 
  38        await Promise.all(a.yrs.map(async (yr:any) => {
  39            const folder = `${a.name} ${a.id}`;
  40            fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/currentYear.json`, yr);
  41            const dc = await getRTCNAActivity(a, yr);           
  42            if(dc.length>0) { 
  43                log.debug(`${a.name}...dc: ${dc.length}`);    
  44                fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/contribution.json`, dc);
  45                return getActivity(a, yr, folder, dc, contr);
  46            }
  47        }))
  48        .catch(err => log.error(`error in ${a.name} ${a.id}: ${err}\n${err.stack}\n${log.inspect(a,1)}`))
  49    ));
  50    getRTCNAcontributions(contr, currentYear);
  51    getRTCNAfunding(contr, currentYear);
  52    log.info(`found ${Object.keys(contr).length} contributions`);
  53    fs.writeJsonFile(`./_data/${currentYear}/contributions.json`, contr);
  54}

  56export function activities(yr:number) {
  57    currentYear = yr;
  58    log.info(`getting ${currentYear} activities`);
  59    activitySteps().catch(log.error);
  60}

  69async function getRTCNAActivity(a:any, yr:any):Promise {
  70    // requests for activity a in year yr:
  71    let dc = await CRIS.get('activity', `activities/${a.id}/years/${yr.id}/department-contributions`);
  72    return (dc.filter && yr.competenceField)? 
  73        dc.filter((r:any)=> r.deptName.match(/RTC.NA/) 
  74                         || r.deptName.match(/RTCHMI/) 
  75                         || yr.competenceField.name==='CW')
  76      : [];
  77}

  79async function getActivity(a: any, yr:any, folder:string, dc:any[], contr:any) {
  80    CRIS.get('activity', `activities/${a.id}/years/${yr.id}/attachments`);
  81    const [vba, commitments, funding, pfp, pc, mc, customers] = await Promise.all([
  82        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/value-benefit-analysis`),
  83        CRIS.get('activity', `activities/${a.id}/fundings/commitments`),
  84        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/financials/fundings`),
  85        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/financials/public-fundings`),
  86        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/costplan/capacity-costs`),
  87        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/costplan/material-costs`),
  88        CRIS.get('activity', `activities/${a.id}/years/${yr.id}/customers`),
  89    ]).catch(log.error); 
  90    log.debug(`${a.name}...details loaded`);    
  91    contr[a.name] = contr[a.name] || []; 
  92    contr[a.name].push(...dc.map((c:any) => { return {
  93        activity:    a,             year:         yr, 
  94        commitments: commitments,   contribution: c,
  95        persCosts:   pc,            matCosts:     mc,
  96        customers:   customers,     funding:      funding,
  97        pfp:         pfp
  98    };}));
  99    log.debug(`${a.name}...found ${contr[a.name].length} contributions`);    
 100    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/vba.json`, vba); 
 101    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/commitments.json`, commitments); 
 102    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/fundings.json`, funding); 
 103    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/publicfundings.json`, pfp); 
 104    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/pc.json`, pc);
 105    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/mc.json`, mc); 
 106    fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/customers.json`, customers); 
 107    return [commitments, funding, pfp, pc, mc, customers];
 108}
