  38async function planningSteps(rtcFilter:boolean) {
  39    getBasicData();
  40    [SPs, COFIs] = await getSPsAndCOFIs();
  41    await getSPplanning(planningList);
  42    await getCOFIplanning(planningList, COFIs);
  43    log.info(`${planningList.length} plannings before`);
  44    planningList = planningList.filter(p=>p.activityNumber.match(`/${currentYear}`));
  45    log.info(`${planningList.length} plannings in ${currentYear}`);
  46    // await getTOPactivities();
  47    fs.writeJsonFile(`./_data/${currentYear}/planning.json`, planningList);
  48    const planning:planningType[] = await getNextYearPlannings(planningList, currentYear);
  49    log.info(`found ${planning.length} plans`);
  50    const table:any[][] = [[
  51        'Prj', 'Title', 'Dept',
  52        'HC', 'Hours', 'PK', 'SK',
  53        'Credit', 'GB', 'Direct', 'Source'
  54    ]];
  55    if (planning && planning.map) { planning.map(p => {
  56        const fundings = {};
  57        if (p.fundings && p.fundings.map) { p.fundings.map((f:PlanFundingType) => {
  58            const type = f.fundingType.name;
  59            const amount = f.value;
  60            const source = f.businessUnit.name;
  61            if (!fundings[type]) { fundings[type] = {amount:0, source:[]}; }
  62            fundings[type].amount += amount;
  63            fundings[type].source.push(`${source} (${amount})`);
  64        });}
  65        p.contribution.map(c => {
  66            const row:any[] = [
  67                p.name,
  68                c.title,
  69                c.deptName,
  70                c.personalCapacity,
  71                c.personalHours,
  72                c.personalCost,
  73                c.materialCost,
  74                fundings['Credit']? fundings['Credit'].amount : '',
  75                fundings['Credit']? fundings['Credit'].source.join('; ') : '',
  76                fundings['Direktmittel']? fundings['Direktmittel'].amount : '',
  77                fundings['Direktmittel']? fundings['Direktmittel'].source.join('; ') : '',
  78            ];
  79            table.push(row);
  80        });
  81    });}
  82    log.info(`writing planning.csv, ${table.length} rows`);
  83    fs.writeTextFile(`./_data/${currentYear}/planning.csv`, table.join('\n'));
  84}

  86export function planning(yr:number, online:boolean) {
  87    const rtcFilter = false;
  88    currentYear = yr;
  89    CRIS.setOnline(online);
  90    log.info(`getting ${currentYear} plannings`);
  91    planningSteps(rtcFilter).catch(log.error);
  92}
