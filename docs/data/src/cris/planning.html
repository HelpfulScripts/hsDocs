  12import * as CRIS                from './CRISrequest';
  13import { log as gLog, fs }      from 'hsnode'; const log = gLog('main');
  14import { getRTCNAcontributions} from './extract';
  15import { getSPsAndCOFIs}        from './basics';
  16
  17const activities:any[] = [];
  18let SPs:any;
  19let COFIs:any;
  20let currentYear = 2018;
  21const rtcContributions = { };
  22
  23export function getPlanning(yr:number) {
  24    currentYear = yr;
  25    log.info(`getting ${currentYear} planning`);
  26    planningSteps().catch(log.error);
  27}
  28
  29async function planningSteps():Promise {
  30    [SPs, COFIs] = await getSPsAndCOFIs();
  31    await getSPplanning();
  32    await getCOFIplanning();
  33    fs.writeJsonFile('./_data/plannings.json', activities);
  34    await getValidActivities();
  35    await getRTCNAcontributions(rtcContributions, currentYear);
  36}
  37
  38async function getSPplanning():Promise {
  39    activities.push(...await CRIS.get('planact', 'planact/planact-numbers?q=SP'));
  40    log.info(`SPs: ${activities.length} planned activities`);  
  41}
  42
  43async function getCOFIplanning():Promise {
  44    await Promise.all(COFIs.filter((cf:any)=>cf.name.length===2).map(searchByCoFi));
  45}
  46
  47async function searchByCoFi(cofi:any):Promise {
  48    const resp = await CRIS.get('planact', `planact/planact-numbers?q=${cofi.name}`);
  49    activities.push(...resp.filter? resp : []);
  50    log.info(`${cofi.name}: ${activities.length} planned activities`);
  51}
  52
  53async function getValidActivities():Promise {
  54    log.info('get activities...');
  55    return await Promise.all(activities.map(getValidActivity));
  56}
  57
  58async function getValidActivity(a:any):Promise {
  59    const folder = `${a.name} ${a.id}`;
  60    const [activity, deptCont, capCosts, matCosts] = await Promise.all([
  61        CRIS.get('planact', `planact/${a.id}/current-version`),
  62        CRIS.get('planact', `planact/${a.id}/department-contributions`),
  63        CRIS.get('planact', `planact/${a.id}/costplan/capacity-costs`),
  64        CRIS.get('planact', `planact/${a.id}/costplan/material-costs`)
  65    ]);
  66    fs.writeJsonFile(`./_data/plannings/${folder}/content.json`, activity);
  67    fs.writeJsonFile(`./_data/plannings/${folder}/department-contributions.json`, deptCont);
  68    fs.writeJsonFile(`./_data/plannings/${folder}/capacity-costs.json`, capCosts);
  69    fs.writeJsonFile(`./_data/plannings/${folder}/material-costs.json`, matCosts);
  70}
