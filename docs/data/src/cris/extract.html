  42function totalCosts(p:contrType, currYr:number) {
  43    const costs = {
  44        lifeTimeCosts: {},
  45        pc:     0,
  46        py:     0,
  47        mc:     0,
  48        depts:  <[string, number][]>[]
  49    };
  50    p.activity.yrs.map((y:ActYear) => {
  51        if(costs.lifeTimeCosts[y.year]===undefined) { costs.lifeTimeCosts[y.year] = 0; }
  52        costs.lifeTimeCosts[y.year] += y.completeCosts;
  53    });
  54    p.persCosts
  55        .filter((cs:any) =>cs.year===currYr)
  56        .map((cs:any) => {
  57            costs.pc += cs.values.costs;
  58            costs.py += cs.values.capacity;
  59            if (cs.values.capacity > 0) { costs.depts.push([cs.competenceSegment.name, cs.values.capacity]); }
  60        });
  61    p.matCosts
  62        .filter((cs:any) =>cs.year===currYr)
  63        .map((cs:any) => {
  64            costs.mc += cs.value;
  65        });
  66    return costs;
  67}

  69function funding(p:contrType, currYr:number) {
  70    const currentYearFunding = {
  71        year:           p.year.year,
  72        totalFunding:   0,
  73        direct:         0,
  74        credit:         0,
  75        publicFunding:  0,
  76        source:         []
  77    };
  78    if (p && p.commitments && p.commitments.map) { 
  79        p.commitments.map((c:any) => {
  80            currentYearFunding.totalFunding += c.value;
  81            currentYearFunding.source.push(c.billingAddress.name);
  82            switch (c.fundingType.name) {
  83                case 'Credit':      currentYearFunding.credit += c.value; break;
  84                case 'Direktmittel':currentYearFunding.direct += c.value; break;
  85            }
  86        });
  87    } else {
  88        log.warn(`${p.activity.name} no commitments`);
  89    }
  90    if (p && p.pfp && p.pfp.filter) { p.pfp
  91        .filter((f:any) => f.year===currYr)
  92        .filter((f:any) => f.rejected===false)
  93        .map((f:any)=> {
  94            currentYearFunding.totalFunding  += f.commitedValue;
  95            currentYearFunding.publicFunding += f.commitedValue;
  96            currentYearFunding.source.push(f.sponsor);
  97        });
  98    } else {
  99        log.warn(`${p.activity.name} no pfp`);
 100    }
 101    return currentYearFunding;
 102}

 104export function getRTCNAcontributions(rtcContributions:{string:contrType}, yr:number) {
 105    const table:any[] = [];
 106    table.push(['Prj','CoFi','Title','ID','Resp','Years',
 107        'Start','End','Lifetime Costs',  
 108        `${yr} PY`,`${yr} PC`,`${yr} MC`,`${yr} Depts`,'Dept','PY','Hours','PC','MC',
 109        'Cost Share', 'Total Funding Share', 'Pfp Share', 'Direct Share', 'Credit Share', 'Source', 'Year', 
 110        'Type', 'Duration'
 111    ]);
 112    log.info(`contributions...${Object.keys(rtcContributions).length}`);
 113    Object.keys(rtcContributions).map(prj =>  
 114        rtcContributions[prj]
 115        .filter((p:contrType) => p.year.year === yr)
 116        .filter((p:contrType) => p.contribution.personalCost+p.contribution.materialCost>0 )
 117        .map((p:contrType) => {
 118            const cyf = funding(p, yr);
 119            const costs = totalCosts(p, yr);
 120            const costShare = (costs.pc + costs.mc)<=0? 0 : (p.contribution.personalCost + p.contribution.materialCost) / (costs.pc + costs.mc);
 121            const title = p.year.currentVersion.titleEnglish || p.year.currentVersion.titleGerman;
 122            const row = [
 123                /* 'Prj'            */ p.activity.name,
 124                /* 'CoFi'           */ p.cofi,
 125                /* 'Title'          */ title? title.trim().replace(/[,|\r|\n]/g,';') : '???',
 126                /* 'ID'             */ p.contribution.id,
 127                /* 'Resp'           */ p.year.responsible.name,
 128                /* 'Years'          */ '>' + Object.keys(costs.lifeTimeCosts).join(' '),
 129                /* 'Start'          */ p.year.currentVersion.start,
 130                /* 'End'            */ p.year.currentVersion.end,
 131                /* 'Lifetime Costs' */ Object.keys(costs.lifeTimeCosts).reduce((acc, yr)=> acc+costs.lifeTimeCosts[yr], 0),
 132                /* 'Total PY'       */ costs.py,
 133                /* 'Total PC'       */ costs.pc,
 134                /* 'Total MC'       */ costs.mc,
 135                /* 'Total Depts'    */ costs.depts
 136                                        .sort((a:[string, number],b:[string, number])=>b[1]-a[1])
 137                                        .map((d:[string, number])=> `${d[0]}(${d[1]})`)
 138                                        .join(' '),
 139                /* 'Dept.'          */ p.contribution.deptName,
 140                /* 'PY'             */ p.contribution.personalCapacity,
 141                /* 'Hours'          */ p.contribution.personalHours,
 142                /* 'PC'             */ p.contribution.personalCost,
 143                /* 'MC'             */ p.contribution.materialCost,
 144                /* 'Cost Share'     */ costShare,
 145                /* 'Total Funding'  */ cyf.totalFunding*costShare, 
 146                /* 'Pfp'            */ cyf.publicFunding*costShare, 
 147                /* 'Direct'         */ cyf.direct*costShare, 
 148                /* 'Credit'.        */ cyf.credit*costShare,
 149                /* 'Source'         */ cyf.source.join(' '),
 150                /* 'Year'           */ cyf.year,
 151                /* 'Type'           */ p.activity.type,
 152                /* 'Duration'       */ ms.toDays(Date.parse(p.year.currentVersion.end) - Date.parse(p.year.currentVersion.start))/365,
 153            ];
 154            table.push(row);
 155        })
 156    );
 157    log.info(`writing departmentContributions.csv, ${table.length} rows`);
 158    fs.writeTextFile(`./_data/${yr}/departmentContributions.csv`, table.join('\n'));
 159}

 228export function charging(rtcContributions:any, yr:number) {
 229    const table:any[] = [];
 230    table.push(['Prj','Title', 'Month', 'Customer', 
 231        'Direct Fnd Dep', 'Type', 'Value', 'Revenue', 'Committed', 'Transfer Price'
 232    ]);
 233    log.info(`chargings...${Object.keys(rtcContributions).length}`);
 234    Object.keys(rtcContributions).map(prj => {
 235        rtcContributions[prj]
 236        .filter((p:any) => p.charging)
 237        .map((p:any) => table.push(...p.charging.filter((c:any) =>  c.accountingMonth.year === yr)
 238            .map((c:any) => [
 239                /* 'Prj'            */ p.activity.name,
 240                /* 'Title'          */ p.year.currentVersion.titleEnglish.trim(),
 241                /* 'Month'          */ c.accountingMonth.month,
 242                /* 'Customer'       */ c.billingArea.name,
 243                /* 'Direct Fnd Dep' */ c.directFundingInformation? c.directFundingInformation.fundingDepartment : '',
 244                /* 'Type'           */ c.moneyType.name.replace(/,/g,';'),
 245                /* 'Value'          */ c.value,
 246                /* 'Revenue'        */ c.revenue,
 247                /* 'Committed'      */ c.yearlyCommitment,
 248                /* 'Transfer Price' */ c.transferPrice,
 249            ])
 250        ));
 251    });
 252    log.info(`writing ${table.length} rows`);
 253    fs.writeTextFile(`./_data/${yr}/charging.csv`, table.join('\n'));
