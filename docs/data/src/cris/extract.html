   4export function rateExtract(resp:any[]) { 
   5    log.info(`${resp.length} rates`);
   6    const rates = { INTERN: {}};
   7    resp.map((d:any) => {
   8        const dep = d.department? d.department.description : d.costCenter.description;
   9        if (!rates[d.type]) { rates[d.type] = {}; }
  10        if (!rates[d.type][dep]) { rates[d.type][dep] = {}; }
  11        rates[d.type][dep][d.year] = d.valueAct;
  12    });
  13    fs.writeJsonFile('./_data/rates.json', rates);
  14
  15    const rateTable = <(string|number)[][]>[];
  16    const nameRow = ['name'];
  17    for (let i=2000; i<=2019; i++) { nameRow.push(''+i); }
  18    rateTable.push(nameRow);
  19
  20    Object.keys(rates.INTERN).map((dep:string) => {
  21        const depRow = [dep];
  22        rateTable.push(depRow);
  23        Object.keys(rates.INTERN[dep]).map(y => {
  24            const yr = parseInt(y)-1999;
  25            if (yr>0) { depRow[yr] = rates.INTERN[dep][y]; }
  26        });
  27    });
  28    fs.writeTextFile('./_data/rateTable.csv', rateTable.map((row)=> row.join(', ')).join('\n'));
