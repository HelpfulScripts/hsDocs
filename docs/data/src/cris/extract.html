  27function funding(p:any, currYr:number) {
  28    const currentYearFunding = {
  29        totalFunding:   0,
  30        direct:         0,
  31        credit:         0,
  32        publicFunding:  0,
  33        source:         []
  34    };
  35    if (p.commitments && p.commitments.map) { 
  36        p.commitments.map((c:any) => {
  37            currentYearFunding.totalFunding += c.value;
  38            currentYearFunding.source.push(c.billingAddress.name);
  39            switch (c.fundingType.name) {
  40                case 'Credit':      currentYearFunding.credit += c.value; break;
  41                case 'Direktmittel':currentYearFunding.direct += c.value; break;
  42            }
  43        });
  44    } else {
  45        log.warn(`no commitments: ${p.activity.name}`);
  46    }
  47    if (p.pfp) { p.pfp
  48        .filter((f:any) => f.year===currYr)
  49        .filter((f:any) => f.rejected===false)
  50        .map((f:any)=> {
  51            currentYearFunding.totalFunding  += f.commitedValue;
  52            currentYearFunding.publicFunding += f.commitedValue;
  53            currentYearFunding.source.push(f.sponsor);
  54        });
  55    }
  56    return currentYearFunding;
  57}
