   3function totalCosts(p:any, currYr:number) {

  10        if(p.lifeTimeCosts[y.year]===undefined) { p.lifeTimeCosts[y.year] = 0; }
  11        p.lifeTimeCosts[y.year] += y.completeCosts;
  12    });

  17            p.py += costs.values.capacity;

  19        });
  20    p.matCosts
  21        .filter((costs:any) =>costs.year===currYr)
