  26async function activitySteps() {
  27    getBasicData();
  28    [SPs, COFIs] = await getSPsAndCOFIs();
  29    await getSPactivities(activityList);
  30    await getCOFIactivities(activityList, COFIs);
  31    // await getTOPactivities();
  32    const current = await getCurrentYearActivities(activityList, currentYear);
  33    const contr = {};
  34    await Promise.all(current.map(async (a:any, i:number, as:any[]) => 
  35            await Promise.all(a.yrs.map(async (yr:any) => {
  36                const folder = `${a.name} ${a.id}`;
  37                const dc = await getRTCNAActivity(a, yr, contr);
  38                if(dc.length>0) { 
  39                    log.debug(`${a.name}...dc: ${dc.length}`);    
  40                    return getMonitoring(a, yr, folder);
  41                }
  42            }))
  43    ));
  44    await charging(contr, currentYear);

  55async function getRTCNAActivity(a:any, yr:any, contr:any):Promise {
  56    const folder = `${a.name} ${a.id}`;
  57    // requests for activity a in year yr:
  58    let dc = await CRIS.get('activity', `activities/${a.id}/years/${yr.id}/department-contributions`);
  59    if (dc.filter) { 
  60        dc = dc.filter((r:any)=> r.deptName.match(/RTC.NA/) 
  61                              || r.deptName.match(/RTCHMI/) 
  62                              || yr.competenceField.name==='CW'
  63        );
  64    }
  65    if(dc.length>0) { 
  66        log.debug(`${a.name}...dc: ${dc.length}`);    
  67        const [obligo, capCosts, planCosts, charging] = await getMonitoring(a.id, currentYear, folder);
  68        contr[a.name] = contr[a.name] || []; 
  69        contr[a.name].push(...dc.map((c:any) => { return {
  70            activity:     a,            year:      yr, 
  71            contribution: c,            obligo:    obligo, 
  72            capCosts:     capCosts,     planCosts: planCosts, 
  73            charging:     charging
  74        };}));
  75        log.debug(`${a.name}...found ${contr[a.name].length} contributions`);    
  76    }
  77}

  79async function getMonitoring(id: number, yr:number, folder:string) {
  80    const [obligo, capCosts, planCosts, charging] = await Promise.all([
  81        CRIS.get('activity', `activities/${id}/monitoring/costs/obligo-internal`),                  // SK orders
  82        CRIS.get('activity', `activities/${id}/monitoring/costs/capacity-costs?accountYear=${yr}`), // charges by person by month
  83        CRIS.get('activity', `activities/${id}/monitoring/costs/plan?year=${yr}`),                  // charges by department by month
  84        CRIS.get('activity', `activities/${id}/monitoring/fundings/chargings?accountYear=${yr}`),   // charges to GB and CR
  85    ]).catch(log.error);
  86    fs.writeJsonFile(`./_data/${yr}/activities/${folder}/obligo.json`, obligo); 
  87    fs.writeJsonFile(`./_data/${yr}/activities/${folder}/capCosts.json`, capCosts); 
  88    fs.writeJsonFile(`./_data/${yr}/activities/${folder}/planCosts.json`, planCosts); 
  89    fs.writeJsonFile(`./_data/${yr}/activities/${folder}/charging.json`, charging); 
  90    return [obligo, capCosts, planCosts, charging];
  91}
