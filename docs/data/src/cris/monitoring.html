  37async function activitySteps(rtcFilter:boolean) {
  38    getBasicData();
  39    [SPs, COFIs] = await getSPsAndCOFIs();
  40    await getSPactivities(activityList);
  41    await getCOFIactivities(activityList, COFIs);
  42    // await getTOPactivities();
  43    fs.writeJsonFile(`./_data/${currentYear}/activities.json`, activityList);
  44    const current:Act[] = await getCurrentYearActivities(activityList, currentYear);
  45    const contr = <{string:monitType}>{};
  46    await Promise.all(current.map(async (a:Act, i:number, as:Act[]) => {
  47        return await Promise.all(a.yrs.map(async (yr:ActYear) => {
  48            const folder = `${a.name} ${a.id}`;
  49            fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/currentYear.json`, yr);
  50            let dc:DepContrType[] = await CRIS.get('activity', `activities/${a.id}/years/${yr.id}/department-contributions`);
  51            if (!dc.filter || !yr.competenceField) { dc = []; }
  52            if (rtcFilter) { dc = getRTCNAActivity(dc, yr.competenceField.name); }       
  53            if(dc.length>0) { 
  54                log.debug(`${a.name}...dc: ${dc.length}`);    
  55                fs.writeJsonFile(`./_data/${currentYear}/activities/${folder}/contribution.json`, dc);
  56                return await getMonitoring(a, currentYear, folder, dc, contr);
  57            }
  58            return dc;
  59        }))
  60        .catch(err => log.error(`error in ${a.name} ${a.id}: ${err}\n${err.stack}\n${log.inspect(a,1)}`));
  61        }
  62    ));
  63    getRTCNAmonitoring(contr, currentYear);
  64    log.info(`found ${Object.keys(contr).length} monitoring entries`);
  65    fs.writeJsonFile(`./_data/${currentYear}/monitoring.json`, contr);
  66}

  68export function monitoring(yr:number, online:boolean) {
  69    const rtcFilter = false;
  70    currentYear = yr;
  71    CRIS.setOnline(online);
  72    log.info(`getting ${currentYear} activities`);
  73    activitySteps(rtcFilter).catch(log.error);
  74}

  85async function getMonitoring(a: Act, yr:number, folder:string, dc:DepContrType[], contr:{string:monitType}) {
  86    const [overview, obligo, capCosts, planCosts, charging] = await Promise.all([
  87        CRIS.get('activity', `activities/${a.id}/monitoring/overview`, cached),                  // SK orders
  88        CRIS.get('activity', `activities/${a.id}/monitoring/costs/obligo-internal`, cached),                  // SK orders
  89        CRIS.get('activity', `activities/${a.id}/monitoring/costs/capacity-costs?accountYear=${yr}`, cached), // charges by person by month
  90        CRIS.get('activity', `activities/${a.id}/monitoring/costs/plan?year=${yr}`, cached),                  // charges by department by month
  91        CRIS.get('activity', `activities/${a.id}/monitoring/fundings/chargings?accountYear=${yr}`, cached),   // charges to GB and CR
  92    ]).catch(log.error);
  93    log.debug(`${a.name}...details loaded`);    
  94    contr[a.name] = contr[a.name] || []; 
  95    contr[a.name].push(...dc.map((c:any):monitType => { return {
  96        activity: a,            overview: overview,
  97        obligo:   obligo,       capCosts:capCosts, 
  98        planCosts:planCosts,    charging:charging
  99    };}));
 100    log.debug(`${a.name}...found ${contr[a.name].length} contributions`);    
 101    fs.writeJsonFile(`./_data/${yr}/activities/${folder}/obligo.json`, obligo); 
 102    fs.writeJsonFile(`./_data/${yr}/activities/${folder}/capCosts.json`, capCosts); 
 103    fs.writeJsonFile(`./_data/${yr}/activities/${folder}/planCosts.json`, planCosts); 
 104    fs.writeJsonFile(`./_data/${yr}/activities/${folder}/charging.json`, charging); 
 105    return [obligo, capCosts, planCosts, charging];
 106}
