  33export function getOnline(url:string, fname:string):Promise {
  34    queueCount++;
  35    return pace.add(() => new Promise((resolve:any, reject:any)=> {
  36        queueCount--;
  37        const options =  {
  38            rejectUnauthorized: false,
  39            url: '',
  40            headers : { 'AuthToken' : apptokens.token}
  41        };
  42        options.url = url;
  43        const d = new Date(PaceTime * queueCount); d.setMinutes(d.getMinutes() + d.getTimezoneOffset());
  44        log.info(`(${pace.getCallingCount()} pending) (${queueCount} - ${date('%h:%mm:%ss',d)} remaining) requesting ${options.url}`);
  45        request(options, (error:any, resp:any, body:string) => {
  46            if(error){
  47                log.error(JSON.stringify(error));
  48                reject(JSON.stringify(error)); 
  49            } else {
  50                if(resp.statusCode !== 200) {
  51                    log.warn(`response code ${resp.statusCode} for ${options.url}`);
  52                }
  53                try {
  54                    const res = JSON.parse(body);
  55                    log.debug(`writing results to ${fname}`);
  56                    fs.writeJsonFile(fname, res); 
  57                    resolve(res);
  58                } catch(err) { 
  59                    log.error(`${body}\n${err}`); 
  60                    resolve({});
  61                }
  62            }
  63        });
  64    }));
  65}

  77    return ((exists && useCached) || offlineOnly)? 
  78        getOffline(fname).catch((err) => {
