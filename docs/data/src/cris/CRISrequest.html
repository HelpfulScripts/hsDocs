  18export function getOnline(module:string, apiCall:string, fname:string):Promise {
  19    queueCount++;
  20    return pace.add(() => new Promise((resolve:any, reject:any)=> {
  21        queueCount--;
  22        const options =  {
  23            rejectUnauthorized: false,
  24            url: '',
  25            headers : { 'AuthToken' : token}
  26        };
  27        options.url = `https://rb-cris.bosch.com/${module}/rest/${apiCall}`;
  28        log.info(`(${queueCount}) requesting ${options.url}`);
  29        reqCount++;
  30        request(options, (error:any, resp:any, body:string) => {
  31            reqCount--;
  32            if(error){
  33                log.error(JSON.stringify(error));
  34                reject(JSON.stringify(error)); 
  35            } else {
  36                if(resp.statusCode !== 200) {
  37                    log.warn(`response code ${resp.statusCode} for ${options.url}`);
  38                }
  39                const res = JSON.parse(body);
  40                log.info(`(${reqCount}) writing results to ${fname}`);
  41                fs.writeJsonFile(fname, res); 
  42                resolve(res);
  43            }
  44        });
  45    }));
  46}
