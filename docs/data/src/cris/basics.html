  53async function getValidActivity(a:any):Promise {
  54    const folder = `${a.name} ${a.id}`;
  55    const activity = await CRIS.get('activity', `activities/${a.id}`);
  56    const yrs = await CRIS.get('activity', `activities/${a.id}/years`);
  57    if (yrs.filter) { // filter '401-unauthroized' messages
  58        const sum:any = {id: a.id, name: a.name, yrs: yrs};
  59        fs.writeJsonFile(`./_data/activities/${folder}/years.json`, yrs);
  60        fs.writeJsonFile(`./_data/activities/${folder}/activity.json`, activity);
  61        fs.writeJsonFile(`./_data/activities/${folder}/summary.json`, sum);
  62        return sum;
  63    } else {
  64        log.warn(`invalid years for ${a.name} (${a.id})`);
  65        fs.writeJsonFile(`./_data/activities/${folder} invalid.json`, activity);
  66        return undefined;
  67    }
  68}

  71async function rateExtract(resp:any[]) { 
  72    log.info(`${resp.length} rates`);
  73    const rates = { INTERN: {}};
  74    resp.map((d:any) => {
  75        const dep = d.department? d.department.description : d.costCenter.description;
  76        if (!rates[d.type]) { rates[d.type] = {}; }
  77        if (!rates[d.type][dep]) { rates[d.type][dep] = {}; }
  78        rates[d.type][dep][d.year] = d.valueAct;
  79    });
  80    fs.writeJsonFile(`./_data/rates.json`, rates);
  81
  82    const rateTable = <(string|number)[][]>[];
  83    const nameRow = ['name'];
  84    for (let i=2000; i<=2019; i++) { nameRow.push(''+i); }
  85    rateTable.push(nameRow);
  86
  87    Object.keys(rates.INTERN).map((dep:string) => {
  88        const depRow = [dep];
  89        rateTable.push(depRow);
  90        Object.keys(rates.INTERN[dep]).map(y => {
  91            const yr = parseInt(y)-1999;
  92            if (yr>0) { depRow[yr] = rates.INTERN[dep][y]; }
  93        });
  94    });
  95    fs.writeTextFile(`./_data/rateTable.csv`, rateTable.map((row)=> row.join(', ')).join('\n'));
  96}
