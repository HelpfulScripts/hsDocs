  14    isProtected?: boolean;

  15    isPublic?: boolean;

  16    isConst?: boolean;

  17    isLet?: boolean;

  18    isOptional?: boolean;

  19    isAbstract?: boolean;

  20    isRest?: boolean;

  21    isConstructorProperty?: boolean;

  22}

  53/**

  54 * generates CSS classes for the provided flags.

  55 * @param mdl the `DocsNode` module for which to provide the css classes

  56 */

  57export function flagsDisplay(mdl:DocsNode):Vnode {

  58    const classMembers = ['Method', 'Property'];

  59    const knownFlags = {

  60        isExported:     'export',

  61        isStatic:       'static',

  62        isPrivate:      'private',

  63        isProtected:    'protected',

  64        isPublic:       'public',

  65        isVar:          'var',

  66        isConst:        'const',

  67        isLet:          'let',

  68        isOptional:     'optional',

  69        isAbstract:     'abstract',

  70        isExternal:     'external',

  71        isRest:         'rest',

  72    };

  73    const flags:string[] = Object.keys(mdl.flags);

  74    const classMember = classMembers.indexOf(mdl.kindString)>=0;

  75    if (classMember) {

  76        if (!mdl.flags.isProtected && !mdl.flags.isPrivate) {

  77            if (flags.indexOf('isPublic')<0) { flags.unshift('isPublic'); }

  78        }

  79    }

  80    return !flags? undefined : flags

  81        .filter(f => f!=='isExported' || !classMember)

  82        .map(f => {

  83            let flag = knownFlags[f] || 'unknown';

  84            return m(`span.hsdocs_flag_${flag}`, flag);

  85        });

  86}
