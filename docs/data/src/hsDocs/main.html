   1import * as CRIS                from './CRISrequest';
   2import { newLog, fs }           from 'hsnode'; const log = newLog('main');
   3import { rateExtract}           from './extract';
   4import { getRTCNAcontributions} from './extract';
   5
   6const activities:any[] = [];
   7const validActivities:any[] = [];
   8const rtcContributions = {};
   9let SPs;
  10let COFIs:any;
  11const currentYear = 2018;
  12
  13Promise.resolve()
  14.then(getBasicData)
  15.then(getSPsAndCOFIs)
  16.then(getSPactivities)
  17.then(getCOFIactivities)
  18.then(() => fs.writeJsonFile('./_data/activities.json', activities))
  19.then(getValidActivities)
  20.then()
  21.then(() => getRTCNAcontributions(rtcContributions))
  22.catch(log.error);

  24function getBasicData() {
  25    return Promise.resolve()
  26    .then(get('core', 'employees/me'))
  27    .then(get('core', 'hourly-rates'))
  28    .then(rateExtract)
  29    .then(get('core', 'operativities'))
  30    // .then((prgs:any[]) => prgs.map(p => get('core', `strategic-programs/${p.id}`)))
  31    .then(get('core', 'role-to-accesslevels'))
  32    .then(get('core', 'profile-attachments'))
  33    ;
  34}
  35
  36function getSPsAndCOFIs() {
  37    return Promise.resolve()
  38    .then(get('core', 'strategic-programs'))
  39    .then((prgs:any[]) => SPs = prgs)
  40    .then(get('core', 'competencefields'))
  41    .then((cofis:any[]) => COFIs = cofis)
  42    ;
  43}

  45/** returns a function that returns a promise to return a CRIS.response */
  46function get(module:string, apiCall:string, useCached=true) {
  47    return () => CRIS.get(module, apiCall, useCached);
  48}

  50function getSPactivities() {
  51    return Promise.resolve()
  52    .then(get('activity', 'activities/activity-numbers?q=SP'))
  53    .then((resp:any[]) => activities.push(...resp))
  54    .then(() => log.info(`SPs: ${activities.length} activities`))  
  55    ;  
  56}
  57
  58function getCOFIactivities() {
  59    return Promise.resolve(COFIs)
  60    .then((cfs:any[]) => Promise.all(cfs.filter(cf=>cf.name.length===2).map(searchByCoFi)))
  61    ;
  62}
  63
  64function searchByCoFi(cofi:any):Promise {
  65    return CRIS.get('activity', `activities/activity-numbers?q=${cofi.name}`)
  66    .then((resp:any) => resp.filter? resp : [])
  67    .then((resp:any[]) => activities.push(...resp))
  68    .then(() => log.info(`${cofi.name}: ${activities.length} activities`));
  69}

  71function getValidActivities():Promise {
  72    return Promise.all(activities.map(getValidActivity))
  73    .then(getCurrentYearActivities)
  74    ;
  75}

  77function getValidActivity(a:any):Promise {
  78    return Promise.resolve()
  79        .then(get('activity', `activities/${a.id}`))
  80        .then((act) => fs.writeJsonFile(`./_data/activities/${a.name}.json`, act))
  81        .then(get('activity', `activities/${a.id}/years`))
  82        .then((yrs:any) => {    // filter '401-unauthroized' messages
  83            if (yrs.filter) { 
  84                fs.writeJsonFile(`./_data/activities/${a.name}/years.json`, yrs);
  85                validActivities.push({id: a.id, name: a.name, yrs: yrs});
  86            } else {
  87                log.warn(`invalid years for ${a.name} (${a.id})`);
  88            }
  89        })    
  90    ;
  91}
  92
  93function getCurrentYearActivities():Promise {
  94    return Promise.all(
  95        validActivities.map((act:any) => Promise.all(act.yrs
  96            .filter((yr:any)=>yr.year===currentYear)
  97            .map((yr:any) => getCurrentYearActivity(act, yr))
  98        ))
  99    );
 100}

 102function getCurrentYearActivity(a:any, yr:any):Promise {
 103    return Promise.resolve()
 104        // requests for activity a in year yr:
 105        .then(get('activity', `activities/${a.id}/years/${yr.id}/attachments`))
 106        .then(get('activity', `activities/${a.id}/years/${yr.id}/department-contributions`))
 107        .then((res:any) => res.filter? res : [])
 108        // .then((res:any[]) => { console.log(`${res.length} ${res.map(r => r.deptName).join(', ')}`); return res;})
 109        .then((res:any[]) => res.filter(r=>r.deptName.match(/RTC.NA/) || r.deptName.match(/RTCHMI/)))
 110        // .then((res:any[]) => { if (res.length>0) { console.log(`     >> ${res.length} ${res.map(r => r.deptName).join(', ')}`);} return res;})
 111        .then((res:any[]) => { if (res.length>0) { rtcContributions[a.name] = rtcContributions[a.name] || []; rtcContributions[a.name].push(...res);} return res;})
 112        // .then((res:any[]) => { if (res.length>0) { console.log(`     >> ${a.name}: ${res.length} ${res.map(r => r.deptName).join(', ')}`);} return res;})
 113        .then(get('activity', `activities/${a.id}/years/${yr.id}/costplan/capacity-costs`))
 114        .then(get('activity', `activities/${a.id}/years/${yr.id}/costplan/material-costs`))
 115        .then(get('activity', `activities/${a.id}/years/${yr.id}/customers`))
 116        // .then(get('activity', `activities/${a.id}/years/${yr.id}/financials/fundings`)) // not authorized
 117    ;
 118}
