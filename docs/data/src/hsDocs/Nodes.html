  50        const node = groupKindTitle[mdl.kind];

  51        if (!node) { log.warn(`unknown kind ${mdl.kindString} (${mdl.kind}) for ${mdl.fullPath}`); }

  52        else { 

  53            // if (DocSets.getNode(mdl.id, mdl.lib)) {

  54            //     log.warn(`node ${mdl.lib}.${mdl.id} already exists`);

  57            return new node(mdl);

  61    static traverse(mdl:json, lib:string, path=''):DocsNode {

  62        mdl.lib = lib;

  63if (lib.indexOf('.')>0) {

  64log.info('');

  65}

  66        mdl.name = mdl.name.replace(/["'](.+)["']|(.+)/g, "$1$2")   // remove quotes 

  67                           .replace(/\//g, '.');                    // "a/b" => "a.b" /

  68        mdl.fullPath = (path==='')? mdl.name : `${path}.${mdl.name}`;

  69    if (DocsNode.errCount>20) { return; }

  70        return DocsNode.makeNode(mdl);

  71    }

 153/**

 154 * establishes a mechanism to check for expected or optional properties, as well as 

 155 * a set of flags and types, for derived classes.

 156 */

 157export class DocsBaseNode extends DocsNode {

 158    mdl:json;

 159    constructor(mdl:json) {

 160        super(mdl);

 161        this.init();

 162        this.set(mdl);

 163        this.setFlags(mdl.flags);

 164    }

 165

 166    init() {

 167        this.FLAGS = flagsNone;

 168    }

 169

 170    set(mdl:json, fields = Object.getOwnPropertyNames(this)) {

 171        fields.filter(f=>this[f]!==undefined).forEach(f => { // expected fields

 172            if (mdl[f]!==undefined) { 

 173                switch(f) {

 174                    case 'type': 

 175                        if (this.TYPE && this.TYPE.indexOf(mdl[f].type)>=0) {

 176                            this[f] = DocsType.makeType(mdl[f], this);

 177                        } else {

 178                            log.warn(`${this.lib} #${mdl.id}: unexpected TYPE '${mdl[f].type}' missing in class '${this.kindString}'`);

 179                            DocsNode.errCount++;

 180                        }

 181                        break;

 182                    case 'comment': 

 183                        this[f] = new DocsComment(mdl[f], this); break;

 184                    case 'children': 

 185                        this[f] = mdl.children.map((c:json) => DocsNode.traverse(c, this.lib, mdl.fullPath)); break;

 186                    case 'signatures':

 187                        this[f] = mdl.signatures.map((c:json) => DocsNode.traverse(c, this.lib, mdl.fullPath)); break;

 188                    case 'indexSignature':

 189                        this[f] = mdl.indexSignature.map((c:json) => DocsNode.traverse(c, this.lib, mdl.fullPath)); break;

 190                    case 'setSignature':

 191                        this[f] = mdl.setSignature.map((c:json) => DocsNode.traverse(c, this.lib, mdl.fullPath)); break;

 192                    case 'getSignature':

 193                        this[f] = mdl.getSignature.map((c:json) => DocsNode.traverse(c, this.lib, mdl.fullPath)); break;

 194                    case 'parameters':

 195                        this[f] = mdl.parameters.map((c:json) => DocsNode.traverse(c, this.lib, mdl.fullPath)); break;

 196                    case 'typeParameter':

 197                        this[f] = mdl.typeParameter.map((c:json) => DocsNode.traverse(c, this.lib, mdl.fullPath)); break;

 198                    default: this[f] = mdl[f];

 199                }

 200            } else if (this.lib && this.kind && !this.kindString) {

 201                log.warn(`${this.lib}: no 'kindString' for ${log.inspect(this, {indent:'   ', colors:null})}`);

 202            } else if (isExpected(this[f]) && isNotIgnored(f)) {

 203                log.warn(`${this.lib}: '${this.kindString}' #${mdl.id}: field '${f}' should be optional`);

 204                DocsNode.errCount++;

 205            }

 206        });

 207        fields = Object.getOwnPropertyNames(mdl);

 208        fields.forEach(f => { // unexpected fields

 209            if (this[f]===undefined && isNotIgnored(f)) {

 210                log.warn(`${this.lib}: '${this.kindString}' #${this.id}: found extra field '${f}'='${log.inspect(mdl[f],{depth:2, indent:'   '})}'`);

 211                DocsNode.errCount++;

 212            }

 213        });

 214    }

 215    

 216    setFlags(mdlFlags:json) {

 217        const staticFlags:Flags = this.FLAGS;

 218        Object.keys(staticFlags).forEach(f => { // expected flags

 219            if (mdlFlags[f]!==undefined) { this.flags[f] = mdlFlags[f]; }

 220            else if (staticFlags[f]===true) {

 221                log.warn(`'${this.kindString}' #${this.id}: flag '${f}' should be optional`);

 222                DocsNode.errCount++;

 223            }

 224        });

 225        const flags = Object.getOwnPropertyNames(mdlFlags);

 226        flags.forEach(f => { // unexpected flags

 227            if (staticFlags[f]===undefined) {

 228                log.warn(`'${this.kindString}' #${this.id}: found extra flag '${f}'='${log.inspect(mdlFlags[f],{depth:1})}'`);

 229                DocsNode.errCount++;

 230            }

 231        });

 232    }
