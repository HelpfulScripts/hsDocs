  10export function flags(mdl:any, ignore:string[]=[]) {

  11    const ignoreExportInKind = ['Method', 'Property'];

  12    const knownFlags = {

  13        isExported:             'export',

  14        isExternal:             'external', // 

  15        isPublic:               'public',

  16        isPrivate:              'private',

  17        isProtected:            'protected',

  18        isConstructorProperty:  'constructorProperty',

  19        isAbstract:             'abstract',

  20        isConst:                'const',

  21        isLet:                  'let',

  22        isVar:                  'var',

  23        isStatic:               'static',

  24        isOptional:             'optional'

  25    };

  26    return m('span.hs-item-flags', !mdl.flags? [] : 

  27        Object.keys(mdl.flags).map((f:string) => {

  28            let ign = false;

  29            let flag = knownFlags[f];

  30            if (flag === undefined) { flag = f; }

  31            else { ign = (ignore.indexOf(flag) >= 0); }

  32            if (flag === 'export' && ignoreExportInKind.indexOf(mdl.kindString)>=0) { ign = true; }

  33            return m(`span.hs-item-${ign?'ignore':(flag===f?'unknown':flag)}-flag`, ign? undefined : flag);

  34        })

  35    );

  36}

  76export function sourceLink(mdl:any) {

  77    const source = mdl.sources? mdl.sources[0] : undefined;

  78    if (source) {

  79        let file = (source.fileName || '').replace('.ts', '.html');

  80        const index = file.indexOf(mdl.lib);

  81        if (index>0) {

  82            file = file.substr(index); // only consider links within the docSet (everything after the lib name)

  83        }

  84        return m('span.hs-item-member-source',  

  85            m('a', { href:`${SourceBase}${mdl.lib}/${file}#${Math.max(0,source.line-5)}`, target:"_blank"}, '[source]')

  86        );

  87    } else {

  88        return m('span.hs-item-member-source', '');

  89    }

 105/**

 106 * creates a function or method signature

 107 */

 108export function signature(s:any, mdl:any): Vnode {

 109    const comma = (i:number) => (i>0)? ', ': '';

 110    function optional(flags: any) {

 111        return (flags && flags.isOptional)? '.hs-item-optional' : '';

 112    }

 113

 114    let sig = [];

 115    if (s) {

 116        if (s.parameters) {

 117            sig = s.parameters.map((p:any, i:number) => m('span', [

 118                comma(i),

 119                m('span.hs-item-sig-param', [

 120                    m(`span.hs-item-name${optional(p.flags)}`, p.name),

 121                    type(p, mdl.lib)

 122                ])

 123            ]));

 124        }

 125        switch (mdl.kindString) {

 126            case undefined: break;

 127            case 'Method':

 128            case 'Function': 

 129            case 'Constructor': 

 130            case 'External module': 

 131            case 'Class': 

 132            case 'Variable': 

 133            case 'Type alias': 

 134            case 'Interface': 

 135                sig.unshift(m('span.hs-item-name', '('));

 136                sig.push(m('span.hs-item-name', ')'));

 137                break;

 138            default: log.warn(`unexpected kindString '${mdl.kindString}'`);

 139        }

 140    }

 141    return m('span.hs-item-signature', sig);

 142}

 156export function type(t:any, lib:string) {

 157    function _type(tt:any):any {

 158        if (!tt) { return ''; }

 159        else { switch (tt.type) {

 160            case undefined:         return '';

 161            case 'array':           return m('span.hs-item-type-array', ['Array<', _type(tt.elementType), '>']);

 162                                    

 163            case 'tuple':           return m('span.hs-item-type-tuple', [

 164                                        '[ ',

 165                                        ...tt.elements.map((e:any, i:number) => [i>0?', ':undefined, _type(e)]),

 166                                        ' ]'

 167                                    ]);

 168            case 'intrinsic':

 169            case 'instrinct':       return m('span.hs-item-type-instrinct', tt.id? libLink('span', lib, tt.fullPath, tt.name) : tt.name); 

 170            case 'stringLiteral':   return m('span.hs-item-type-string-literal', tt.type); 

 171            case 'union':           return m('span.hs-item-type-union', [...tt.types.map((e:any, i:number) => [i>0?' | ':undefined, _type(e)])]);

 172            case 'reference':       let refRes = tt.name;

 173                                    if (tt.id) {

 174                                        const typeRef = DocSets.get(lib, tt.id);

 175                                        if (typeRef.typeArguments) { refRes = typeRef.name+'<'+ typeRef.typeArguments.map(_type).join(', ') + '>'; }

 176                                        else if (typeRef.id)       { refRes = libLink('a', lib, typeRef.fullPath, typeRef.name); }

 177                                        else                       { refRes = typeRef.name; }

 178                                    }

 179                                    return m('span.hs-item-type-reference', refRes);

 180            case 'reflection':      let rflRes;

 181                                    if (tt.declaration) {

 182                                        rflRes = !tt.declaration.children? tt.declaration.kindString :

 183                                            m('span.hs-item-reflection', [

 184                                                '{ ',

 185                                                ...tt.declaration.children.map((c:any, i:number) => 

 186                                                    [i>0?', ':undefined, c.name, ': ', _type(c.type)]

 187                                                ),

 188                                                ' }'

 189                                            ]);

 190                                    } else {

 191                                        rflRes = 'UNKNOWN';

 192                                    }

 193                                    return m('span.hs-item-type-reflection', rflRes);

 194            default: log.warn('unknown type '+ tt.type);

 195                     log.warn(log.inspect(t,3));

 196                     return t.type;

 197        }}

 198    }

 199

 200    try {

 201       return m('span', !t.type? '': [

 202           m('span.hs-item-name',':'), 

 203           m('span.hs-item-sig-type', _type(t.type)),

 204           defaultVal(t, lib)

 205        ]);

 206    } catch(e) { log.error(e); log.error(e.trace); }

 207}
