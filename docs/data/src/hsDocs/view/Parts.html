  10export function flags(mdl:any, ignore:string[]=[]) {

  11    const ignoreExportInKind = ['Method', 'Property'];

  12    const knownFlags = {

  13        isExported:             'export',

  14        isExternal:             'external', // 

  15        isPublic:               'public',

  16        isPrivate:              'private',

  17        isProtected:            'protected',

  18        isConstructorProperty:  'constructorProperty',

  19        isAbstract:             'abstract',

  20        isConst:                'const',

  21        isLet:                  'let',

  22        isVar:                  'var',

  23        isStatic:               'static',

  24        isOptional:             'optional'

  25    };

  26    return m('span.hs-item-flags', !mdl.flags? [] : 

  27        Object.keys(mdl.flags).map((f:string) => {

  28            let ign = false;

  29            let flag = knownFlags[f];

  30            if (flag === undefined) { flag = f; }

  31            else { ign = (ignore.indexOf(flag) >= 0); }

  32            if (flag === 'export' && ignoreExportInKind.indexOf(mdl.kindString)>=0) { ign = true; }

  33            return m(`span.hs-item-${ign?'ignore':(flag===f?'unknown':flag)}-flag`, ign? undefined : flag);

  34        })

  35    );

  36}

 119export function sourceLink(mdl:any) {

 120    const source = mdl.sources? mdl.sources[0] : undefined;

 121    if (source) {

 122        let file = (source.fileName || '').replace('.ts', '.html');

 123        const index = file.indexOf(mdl.lib);

 124        if (index>0) {

 125            file = file.substr(index); // only consider links within the docSet (everything after the lib name)

 126        }

 127        return m('span.hs-item-member-source',  

 128            m('a', { href:`${SourceBase}${mdl.lib}/${file}#${Math.max(0,source.line-5)}`, target:"_blank"}, '[source]')

 129        );

 130    } else {

 131        return m('span.hs-item-member-source', '');

 132    }

 148/**

 149 * creates a function or method signature

 150 */

 151export function signature(s:any, mdl:any): Vnode {

 152    function params(s:any) {

 153        return s.parameters.map((p:any, i:number) => m('span', [

 154            comma(i),

 155            m('span.hs-item-sig-param', [

 156                m(`span.hs-item-name${optional(p.flags)}`, p.name),

 157                type(p, mdl.lib)

 158            ])

 159        ]));

 160    }

 161    const comma = (i:number) => (i>0)? ', ': '';

 162    const optional = (flags:any) => (flags && flags.isOptional)? '.hs-item-optional' : '';

 163

 164    let sig = [];

 165    if (s) {

 166        if (s.parameters) { sig = params(s); }     

 167        else if (s.type && s.type.declaration && s.type.declaration.indexSignature) {

 168            sig = s.type.declaration.indexSignature.map((isig:any) => m('span', params(isig)));

 169        }

 170        else if (s.type && s.type.declaration && s.type.declaration.signatures) {

 171            sig = s.type.declaration.signatures.map((isig:any) => m('span', params(isig)));

 172        }

 173        switch (mdl.kindString) {

 174            // unmodified:

 175            case 'Variable': 

 176            case 'Parameter': 

 177            case 'Property': 

 178            case 'Type alias': 

 179            case 'Interface': 

 180            case 'Class': 

 181            case 'External module': 

 182            case 'Enumeration': 

 183            case 'Enumeration member': 

 184                break;

 185            // enclose in (...):

 186            case undefined: break;

 187            case 'Method':

 188            case 'Accessor':

 189            case 'Function': 

 190            case 'Constructor': 

 191            case 'Object literal': 

 192                sig.unshift(m('span.hs-item-name', '('));

 193                sig.push(m('span.hs-item-name', ')'));

 194                break;

 195            default: log.warn(`unexpected kindString '${mdl.kindString}'`);

 196        }

 197    }

 198    return m('span.hs-item-signature', sig);

 199}

 213export function type(t:any, lib:string) {

 214    function _type(tt:any):any {

 215        if (!tt) { return ''; }

 216        else { switch (tt.type) {

 217            case undefined:         return '';

 218            case 'array':           return m('span.hs-item-type-array', ['Array<', _type(tt.elementType), '>']);

 219                                    

 220            case 'tuple':           return m('span.hs-item-type-tuple', [

 221                                        '[ ',

 222                                        ...tt.elements.map((e:any, i:number) => [i>0?', ':undefined, _type(e)]),

 223                                        ' ]'

 224                                    ]);

 225            case 'typeParameter':   // found in d3.js

 226                                    // log.warn(`found 'typeParameter', lib=${lib}, structure:\n${log.inspect(tt, 1)}`);

 227            case 'intrinsic':

 228            case 'instrinct':       return m('span.hs-item-type-instrinct', tt.id? libLink('span', lib, tt.fullPath, tt.name) : tt.name); 

 229            case 'stringLiteral':   return m('span.hs-item-type-string-literal', tt.type); 

 230            case 'union':           return m('span.hs-item-type-union', [...tt.types.map((e:any, i:number) => [i>0?' | ':undefined, _type(e)])]);

 231            case 'reference':       let refRes = tt.name;

 232                                    if (tt.id) {

 233                                        const typeRef = DocSets.get(lib, tt.id);

 234                                        if (typeRef.typeArguments) { refRes = typeRef.name+'<'+ typeRef.typeArguments.map(_type).join(', ') + '>'; }

 235                                        else if (typeRef.id)       { refRes = libLink('a', lib, typeRef.fullPath, typeRef.name); }

 236                                        else                       { refRes = typeRef.name; }

 237                                    }

 238                                    return m('span.hs-item-type-reference', refRes);

 239            case 'reflection':      let rflRes;

 240                                    if (tt.declaration) {

 241                                        rflRes = !tt.declaration.children? tt.declaration.kindString :

 242                                            m('span.hs-item-reflection', [

 243                                                '{ ',

 244                                                ...tt.declaration.children.map((c:any, i:number) => 

 245                                                    [i>0?', ':undefined, c.name, ': ', _type(c.type)]

 246                                                ),

 247                                                ' }'

 248                                            ]);

 249                                    } else {

 250                                        rflRes = 'UNKNOWN';

 251                                    }

 252                                    return m('span.hs-item-type-reflection', rflRes);

 253            default: log.warn('unknown type '+ tt.type);

 254                     return m('span.error', tt.type);

 255        }}

 256    }

 257

 258    try {

 259       return m('span', !t.type? '': [

 260           m('span.hs-item-name',':'), 

 261           m('span.hs-item-sig-type', _type(t.type)),

 262           defaultVal(t, lib)

 263        ]);

 264    } catch(e) { log.error(e); log.error(e.trace); }

 265}
