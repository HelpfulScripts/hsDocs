  10/**

  11 * Main comment processing. The result appears directly below the title in the main panel.

  12 * Function parameters are not reported in short form here since it is assumed they will be listed 

  13 * individually below the main comment

  14 * @param mdl the module to scan for comments

  15 * @return a vnode representing the comment entries

  16 */

  17export function commentLong(mdl:any):Vnode { 

  18    let content:any[] = [];

  19    if (mdl.comment) {

  20        content.push(textOrShortTextOrDescription(mdl.comment, false));

  21        content.push(returns(mdl.comment, false));

  22        content.push(commentRemainder(mdl.comment));

  23    }

  24    return m('.hs-item-comment', content);

  25}

  27/**

  28 * Shortended comment processing. This form is used to report on subitems below the main panel item.

  29 * If `short` is true, only the first paragraph of the main comment will be returned. Otherwise, 

  30 * this function creates a full comment including an inline list of parameters and the return value

  31 * @param mdl the module to scan for comments

  32 * @param short if true, only the first paragraph of the main comment is processed.

  33 * @return a vnode representing the comment entries

  34 */

  35export function comment(mdl:any, short=false):Vnode {

  36    let content:any[] = [];

  37    if (mdl.comment) {

  38        content.push(textOrShortTextOrDescription(mdl.comment, short));

  39        if (!short) {

  40            content.push(otherCommentTags(mdl.comment));

  41            if (mdl.parameters) {

  42                content = content.concat(mainCommentParams(mdl.parameters));

  43            }

  44        }

  45        content.push(returns(mdl.comment, false));

  46        content.push(commentRemainder(mdl.comment));

  47    }

  48    return m('.hs-item-comment', content);

  49}

  51/**

  52 * Report the item's description. This can come in different forms that are all handled here:

  53 * - comment.shortText: 

  54 * - comment.text: 

  55 * - comment.tags[{tag:'description}, text:]

  56 * Any resulting comment will be translated from markdown to html and returned as a `Vnode`.

  57 * @param comment the comment object to parse

  58 * @param short boolean; if true, only the first paragraph of the description will be returned

  59 */

  60function textOrShortTextOrDescription(comment:any, short:boolean):Vnode {

  61    let text = (comment.shortText || '');

  62    if (comment.text) { text += '\n'+ (comment.text || ''); }

  63    if (comment.tags) {

  64        comment.tags.map((tag:any) => {

  65            switch (tag.tag) {

  66                case 'shortText':

  67                case 'description': text = `${text}\n${tag.text}`; break;

  68                case 'param': text = `${text}\n- **${tag.param}**: ${tag.text}`; break;

  69            }

  70        });

  71    }

  72    // search for pattern 

  73    text = text.replace(//gi, short? '' : example);

  74    return m('.hs-item-comment-desc', prettifyCode(text, short));

  75}

  77/**

  78 * creates the `returns` message for a function or method.

  79 */

  80function returns(comment:any, short:boolean):Vnode {

  81    let text = comment.returns;

  82    return m('.hs-item-comment-return', !text? '': [            

  83        m('span.hs-item-comment-tag', 'returns:'), 

  84        m('span.hs-item-comment-text', text)

  85    ]);

  86}

  88function commentRemainder(comment:any):string|Vnode {

  89    return m('', Object.keys(comment).map((tag:any) => {

  90            switch(tag) {

  91                case 'tags':        // already handled

  92                case 'shortText':   // already handled

  93                case 'text':        // already handled

  94                case 'description': // already handled

  95                case 'returns':     // already handled

  96                        return '';

  97                default: return m('.hs-item-comment-special', [

  98                    m('span.hs-item-comment-tag', tag), 

  99                    m('span.hs-item-comment-text', comment[tag])

 100                ]);

 101            }

 102    }));

 103}

 105function otherCommentTags(comment:any):string|Vnode {

 106    return m('', !comment.tags? [] : comment.tags.map((tag:any) => {

 107        switch(tag.tag) {

 108            case 'description': return ''; // skip since already handled

 109            default: return m('.hs-item-comment-special', [

 110                m('span.hs-item-comment-tag', tag.tag), 

 111                m('span.hs-item-comment-text', tag.text)

 112            ]);

 113        }

 114    }));

 115}

 117function mainCommentParams(params:any):Vnode {

 118    return m('.hs-item-comment-params',  params.map((par:any) =>

 119        m('.hs-item-comment-param', [

 120            m('span.hs-item-comment-tag', par.name+':'), 

 121            m('span.hs-item-comment-text', !par.comment? '' :

 122                ((par.defaultValue!==undefined)? `[default: ${par.defaultValue}] ` : '') + par.comment.text

 123            )

 124        ])

 125    ));

 126}

 128/**

 129 * finds segments of `...` in `comment` and replaces them with a prettified version.

 130 * Currently the function performs operations in this sequence:

 131 * - add indentation for brackets {...}

 132 * - wrap the <code>...</code> part within <pre>...</pre> brackets

 133 * - apply markdown syntax

 134 * - wrap results in a trusted mithril node

 135 * @param comment the comment comment 

 136 * @param short only return the first paragraph

 137 */

 138function prettifyCode(comment:string, short:boolean):Vnode { 

 139//    const indentSpaces = 2;

 140    let result = comment;

 141

 142    function braceIndenting(text:string): string {

 143        let indent = 0;

 144        const result = text

 145            .substring(6, text.length-7)    // remove  and 

 146            .trim()

 147            .replace(/(<)/g, '<').replace(/(>)/g, '>')

 148            .split('\n')

 149            .map((l:string) => {

 150                let oldIndent = indent;

 151                let k = l.trim();

 152                indent += Math.max(-1, Math.min(1, k.split('{').length - k.split('}').length)); 

 153                indent += Math.max(-1, Math.min(1, k.split('[').length - k.split(']').length)); 

 154                return '"hs-code-indent">'.repeat(((indent < oldIndent)?indent:oldIndent)) + k;

 155            })

 156            .join('\n')

 157            .trim();

 158        return '

' + result + '
';

 159    }

 160

 161    result = result.replace(/([\S\s]*?)<\/code>/gi, braceIndenting);

 162    return m.trust(markDown(result, short, m.route.get()));

 163}
