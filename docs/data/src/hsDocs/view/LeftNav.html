  14 * Constructs the left-hand navigation pane

  16export class LeftNav extends Layout {

  17    getComponents(node: Vnode): Vnode {

  18        let lib = m.route.param('lib');

  19        let field = m.route.param('field');

  20        const docSet = DocSets.get(lib, 0) || {name:'unknown', id:0};

  21        return m('.hs-left', [m('.hs-left-nav', navList(docSet, field))]);

  22    }

  25/** creates the list if modules (`*.ts` files) */

  26function navList(docSet:any, field:string):Vnode[] {

  27    /** */

  28    function collectModules(docSet:any) {

  29        const modulesByName = {};

  30        docSet.modules = [];

  31        if (docSet.children) {

  32            docSet.children.forEach((c:any) => {

  33                // don't show modules from other projects (isExternal flag) or modules on the ignore list

  34                if (!(c.flags && c.flags.isExternal) && !ignoreModules[c.name]) {

  35                    const name = c.innerModule? c.innerModule : c.name;

  36                    let module = modulesByName[name];

  37                    if (!module) {  // new module

  38                        docSet.modules.push(module = modulesByName[name] = { 

  39                            name: name,

  40                            lib: docSet.lib,

  41                            fullPath: docSet.fullPath + '.'+ name,

  42                            groups:[]

  43                        });

  44                    }

  45                    // get existing module groups

  46                    const groups = {};

  47                    module.groups.forEach((g:any) => groups[g.title] = g); 

  48                    // for each group in child:

  49                    if (c.groups) { c.groups.forEach((g:any) => {

  50                        let group = groups[g.title]; // get existing 

  51                        if (!group) {                 //  else create new

  52                            group = groups[g.title] = {

  53                                children:[],

  54                                kind: g.kind,

  55                                title: g.title

  56                            };

  57                        module.groups.push(group);

  58                        }

  59                        group.children = group.children.concat(g.children);

  60                    });}

  61                }

  62            });

  63        }

  64    }

  65    /** 

  66     * processes a module, i.e. a `.ts` file.

  67     */

  68    function externalModule(mdl:any) {

  69        let selected = false;

  70        if (field===''+mdl.id || field.indexOf(mdl.fullPath) === 0) { selected=true; }

  71

  72        return m(Collapsible, {css:`.hs-left-nav-module`, preArrow: true, isExpanded:selected, components:[

  73            m(`${selected?'.hs-left-nav-selected':''}`, libLink(`a.hs-left-nav-module-name `, mdl.lib, mdl.fullPath, mdl.name)),

  74            !mdl.groups? undefined : mdl.groups.map((g:any) => entries(g, mdl, field))

  75        ]});

  76    }

  77

  78    if (docSet.kind === 0) { // External DocSets

  79        collectModules(docSet);

  80//        const modules = docSet.children? docSet.children.map(externalModule) : ['no children'];

  81        const modules = docSet.modules.map(externalModule);

  82        modules.unshift(m('.hs-left-nav-header', 'Modules'));

  83        return [m('.hs-left-nav-content', modules)];

  84    }

  85}

  88 * modules to ignore in the list

  89 */

  90const ignoreModules = {

  91    overview:   true,   // the project overview.ts file

  92    index:      true    // the index.ts file

  93};

  94

  95//const expanded: {string?:boolean} = {};

  96

  97/**

  98 * processes a group of entries, e.g. Variables, Functions, or Classes.

  99 * @param group the group structure within the docset

 100 * @param mdl the entire module docset

 101 * @param field the field ID to be displayed on the main panel

 102 */

 103function entries(group:any, mdl:any, field:string) {

 104    function moduleGet(c:any) {

 105        return DocSets.get(mdl.lib, c);

 106    }

 107    /**

 108     * processes one entry within a group, e.g. one variable, function, or class.

 109     */

 110    function entry(mod:any) { 

 111        const selected = (field===''+mod.id || field===mod.fullPath)? '.hs-left-nav-selected' : '';

 112        const exported = (mod.flags && mod.flags.isExported);

 113        const statik   = (mod.flags && mod.flags.isStatic);

 114        const css = `a.hs-left-nav-entry ${selected} ${exported?'.hs-left-nav-exported' : ''}`;

 115        return (!exported && group.title==='Variables')? '' :   // ignore local module variables

 116            m('', [

 117                statik? 'static': '',

 118                libLink(css, mod.lib, mod.fullPath, mod.name)

 119            ]);

 120    }

 121

 122    function empty(mod:any) { return mod !== ''; }

 123

 124    let grp = [];

 125    if (group && group.children) {

 126        grp = group.children

 127            .map(moduleGet)         // replace id reference by module

 128            .sort(exportAscending)  // sort: exported first, then alphabetically

 129            .map(entry)             // replace module by vnode structure

 130            .filter(empty);         // filter empty elements

 131        if (grp.length > 0) { // add an entries header if there are elements

 132            grp.unshift(m('.hs-left-nav-header', group.title));

 133        }

 134    }

 135    return (grp.length > 1)? m(`.hs-left-nav-entries`, grp) : '';

 136}

 138/**

 139 * sorting function: sort first by exported status, then alphabetically.

 140 */

 141function exportAscending(a:any, b:any):boolean|number {

 142    if (a.flags && b.flags) {

 143        if (a.flags.isExported && b.flags.isExported) { return a.name > b.name; }

 144        else if (a.flags.isExported) { return -1; }

 145        else if (b.flags.isExported) { return 1; }

 146        else { return a.name > b.name; }

 147    } else { return a.name > b.name; }

 148}
