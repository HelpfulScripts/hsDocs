   1/**

   2 * Comment sections may contain code examples that demonstrate code behaviour and usage.

   3 * 

   4 * `hsDocs` will interpret and execute Javascript instructions within a `'script.js'>` tag.

   5 * and stylesheet instructions with a <`file name='style.css'`> tag, as in following example (also see Example below):

   6 * 

   7 *     <example> 

   8 *     'script.js'>

   9 *     m.mount(root, { 

  10 *         view:() => m(hsLayout.Layout, { columns:[], 

  11 *             content:['first line','second line')]

  12 *         })

  13 *     });

  14 *     

  15 * 

  16 *     'style.css'>

  17 *     .hsLeaf { 

  18 *         color: blue; 

  19 *     }

  20 *     

  21 *     </example>

  22 *  

  23 * 

  24 * ### Scripts 

  25 * Scripts are enclosed by `'script.js'>` and `` tags and are intended to be live examples for 

  26 * explaining some concept of the module are contained in. 

  27 * 

  28 * hsDocs internally uses the [global Function object][Function] to parse and execute the script. 

  29 * Thus the script runs in the global scope and has access to global objects. A number of additional objects are provided

  30 * as parameters in the `Function` constructor and are thus available to the script. 

  31 * hsDocs currently provides the following namespaces as parameters

  32 * for use in the scripts:

  33 * - **m**: the `Mithril` m function    

  34 * - **layout**: the {@link hsLayout: `hsLayout`} namespace, providing functions to layout the browser window.

  35 * - **widget**: the {@link hsGraph: `hsGraph`} namespace, providing various UI widget functions.

  36 * - **root**: the DOM element to attach content to.

  37 * 

  38 * Scripts can use and manipulate any of the provided objects. 

  39 * #### Using Mithril:

  40 * mount a `mithril Vnode` on the provided root DOM element using `m.mount` or `m.render`. 

  41 * Do not use `m.route` as only a single call is allowed per web app and that is used to manage the 

  42 * main hsDocs site menu and navigation.

  43 * ```

  44 * m.mount(root, { view:() => m('div', 'content')});

  45 * ```

  46 * #### Using d3:

  47 * ```

  48 * const base = d3.select(root);

  49 *  const svg = base.append('svg')

  50 *     .classed('baseSVG', true)

  51 *     .attr('height', '100%')

  52 *     .attr('width', '100%')

  53 *     .attr('preserveAspectRatio', 'xMinYMin meet');

  54 * ```

  55 * 

  56 * #### Configuring the `example`

  57 * The `<example>` accepts additional arguments to configure the example:

  58 * - height=px   the height of the example box; defaults to 300

  59 * - libs={:}   a list of additional libraries to load. `export name` is the symbol exported by the library.

  60 * This symbol will be available to the script. The main intent is to load the library being documented so that its features 

  61 * can be illustrated by the example script.

  62 * For example:

  63 * 

  64 * <example height=500px libs={hsGraphd3:'hsGraphd3',d3:'https://d3js.org/d3.v5.min.js'}>

  65 * 

  66 * If `path` is a module nane (e.g. 'hsDocs'), 

  67 * this involves the following attempts

  68 * at resolving the library path:

  69 * 1. load from 'node_modules//.js, relative to the current web page.

  70 * 2. load from 'https://helpfulscripts.github.io//.js'

  71 * If `path` is structured, i.e. contains '/', it is requested as is. 

  72 * 

  73 * [Function]:https://developer.mozilla.org/en-US/docs/Web/JavaScript/Reference/Global_Objects/Function

  74 * 

  75 * ### Styles

  76 * Styles will be automatically sandboxed so they are valid only within the enclosing example tags. 

  77 * This is achieved by prefixing css tags with a unique exmple ID and allows multiple examples to co-exist on the same page.

  78 * In the DOM, the example ID is assgned top the <example> tag.

  79 * Use `$exampleID` as css tag in the `css` section of the example to refer to the <example> element, 

  80 * as shown below.

  81 * 

  82 * ### Example

  83 * 

  84 * script.js'>

  85 * m.mount(root, { 

  86 *     view:() => m(hsLayout.Layout, {

  87 *         css:'.myExample', 

  88 *         columns:[], 

  89 *         content:[

  90 *             'left column',

  91 *             'right column'

  92 *         ]

  93 *     })

  94 * });

  95 * 

  96 * style.css'>

  97 * $exampleID { height: 200px;}

  98 * .hs-layout { 

  99 *     margin:0; 

 100 * }

 101 * .myExample { 

 102 *     color: red; 

 103 *     font-weight:bold; 

 104 * }

 105 * 

 106 *

 154/**

 155 * creates the example configuration, generates the DOM hook, and sets up the example execution.

 156 * `example` takes a context map of the form `{ name:module, ...}` containing libraries to be used 

 157 * within the example, and returns a function that can be used in calls to `string.replace`. 

 158 * See the following example:

 159 * 



 160 * import * as layout from "layout";


 161 * text = text.replace(/([\S\s]*?)<\/example>/gi, example({layout:layout}));


 162 * 

 163 * The modules `m`, `hsLayout`, and `hsGraph` will be added by default as 

 164 * ` { m: m, layout: layout, widget: widget } `

 165 * @param exmpl the example string extracted from the comment, including the `` tags.

 166 * @param context the context in which the example script should be run, expressed as `name`:`namespace` pairs.

 167 */

 168export function example(exmpl:string) { 

 169    const instance = shortCheckSum(exmpl);

 170    let IDs = gInitialized[instance]; 

 171    if (!IDs) {

 172        IDs = gInitialized[instance] = initDesc(IDs);

 173        IDs.executeSource = exmpl;

 174        createExecuteScript(IDs, exmpl);

 175    }

 176    if (!document.getElementById(IDs.menuID)) { 

 177        addExampleStructure(IDs);

 178        delay(1)(IDs)

 179            .then(executeScript)

 180            .catch(executeError);

 181    }

 182

 183    const frameHeight = (IDs.attrs? IDs.attrs.height : undefined) || '300px';

 184    const wrapWithID = (css:string) => css==='$exampleID'? `#${IDs.exampleID}`: `#${IDs.menuID} ${css}`;

 185

 186    // prefix css selectors with ID of main execution area to sandbox the scope

 187    // (^|}): start of string or end of previous style def

 188    // \s*?: any white spaces

 189    // (\S*?): capturing group: css name

 190    // \s*?{: whitespaces, followed by start of style def

 191    const style = (!IDs.pages['css'])? '':                              // empty if no css defined

 192                    IDs.pages['css'].replace(/(^|})\s*?(\S*?)\s*?{/gi,    // otherwise wrap each css statement

 193                        (x:string, ...args:any[]) => x.replace(args[1], wrapWithID(args[1]))

 194    );

 195    return `"height:${frameHeight}" class="hs-layout-frame">`;

 196}

 198async function createExecuteScript(IDs:CommentDescriptor, exmpl:string): Promise {

 199    try {

 200        const desc = await getCommentDescriptor(IDs, exmpl);

 201        IDs.runScript = (root:any) => {

 202            const libNames = Object.keys(modules);

 203            const missingLibs = IDs.attrs.libs.map(l => l.sym).filter(l => libNames.indexOf(l)<0);

 204            if (missingLibs.length > 0) {

 205                log.info(`expected modules to be loaded: ${missingLibs.join(', ')}`);

 206            } else {

 207                const libs = libNames.map(name => modules[name]);

 208                const scriptFn = new Function('root', ...libNames, desc);    

 209                log.info('running example script');

 210                scriptFn(root, ...libs);

 211            }

 212        };

 213        log.debug('example function ready');

 214        IDs.created = true;

 215        return true;

 216    } catch(e) { 

 217        log.error(`creating script: ${e}\n${e.stack}`); 

 218        return false;

 219    } 

 220}

 288/**

 289 * execute the provided script 

 290 * @param IDs provides the context in which the script is exceuted/

 291 */

 292function executeScript(IDs:CommentDescriptor) {

 293    const root = document.getElementById(IDs.menuID);

 294    if (root && IDs.created) {

 295        try { IDs.runScript(root); }

 296        catch(e) { 

 297            log.error(`error executing script: ${e}\n${IDs.executeSource}\n${e.stack}`); 

 298        }

 299    } else {

 300        // log.error(`root not found for menuID ${IDs.menuID}`);

 301    }

 302    m.redraw();

 303    return IDs;

 304}

 310/**

 311 * Fills in all fields of the CommentDescriptor provided as `IDs`.

 312 * @param IDs the CommentDescriptor to complete

 313 * @param example the example string, including  tag

 314 * @return the script to execute, as a string

 315 */

 316async function getCommentDescriptor(IDs:CommentDescriptor, example:string):Promise {

 317    let result = '';

 318    

 319    // let attrs = example.match(/)/gi);

 320    let attrs = example.match(//); // capture all parameters to `example`

 321    if (attrs && attrs[1]) { 

 322        findTokens(IDs, attrs[1]);

 323        if (IDs.attrs.libs.length>0) {

 324            await Promise.all(IDs.attrs.libs.map(async l => {

 325                log.debug(`loading lib: ${l.sym}: ${l.path}`);

 326                const content = await loadScript(l.sym, l.path);

 327                if (content) { modules[l.sym] = content[l.sym]; }

 328            }));

 329        }

 330        log.debug(`found attrs:\n ${log.inspect(IDs.attrs, 5)}`);

 331    }

 332    example.replace(/|"]>([\S\s]*?)<\/file>/gi, function(text:string) {

 333        const args = [...arguments];

 334        const content = args[2].trim(); // the part between  and 

 335        IDs.desc.items.push(args[1]);   // record available extensions, i.e. 'js', 'html', etc 

 336        IDs.pages[args[1]] = content;   // associate the content with the extension

 337        return result;

 338    });

 339    return IDs.pages['js'];

 340}

 342function findTokens(IDs:CommentDescriptor, str:string) {

 343    const results = [];

 344    // find 'key=value' and 'key={value}' statements

 345    const reg = new RegExp(/((\w+?)\s*?=\s*?([^\s{},]+|{[\s\S]+?}))/g);

 346    let r;

 347    while(r = reg.exec(str)) {

 348        results.push({key:r[2], val:r[3]});

 349    }

 350    results.map(kv => {

 351        switch (kv.key) {

 352            case 'height': IDs.attrs['height'] = kv.val; break;       

 353            case 'libs':    // {graph:'/node_modules/hsgraphd3/index.js'}

 354                const ligreg = new RegExp(/((\w+?)\s*:\s*([^\s,}]+))/g);

 355                while(r = ligreg.exec(kv.val)) {

 356                    IDs.attrs.libs.push({sym:r[2],path:r[3].replace(/['"]/g, '')});

 357                }

 358        }

 359    });

 360    log.debug(`${str} --> \n${log.inspect(IDs.attrs,5)}`);

 361}

 364/**

 365 * attempts to load a library for the specified `path`. If `path` is a module nane (e.g. 'hsDocs'), 

 366 * this involves the following attempts

 367 * at resolving the library path:

 368 * 1. load from 'node_modules//.js, relative to the current web page.

 369 * 2. load from 'https://helpfulscripts.github.io//.js'

 370 * If `path` is structured, i.e. contains '/', it is requested as is. 

 371 * @param path module name, such as 'hsDocs'

 372 */

 373async function loadScript(sym:string, path:string) {

 374    const load = (p:string) => m.request({ method: "GET", url: p, extract: (xhr:any) => xhr });

 375

 376    const paths = [

 377        `node_modules/${path.toLowerCase()}/${path}.js`,

 378        `https://helpfulscripts.github.io/${path}/${path}.js`

 379    ];

 380    let content:any;

 381    try {

 382        if (path.indexOf('/')>=0) {     // if structured: call as is

 383            content = await load(path);

 384        } else {                        // else try other paths

 385            for (let i=0; i<=paths.length; i++) {

 386                try { 

 387                    content = await load(paths[i]); 

 388                    break;

 389                }

 390                catch(e) {}

 391            }

 392            log.debug(`${path} ${content?'found':'not found'}`);

 393        }

 394    } catch(e) { log.error(`loading lib ${path}`);}

 395    try { 

 396        const code:string = content.responseText;

 397        if (code.startsWith('this["')) {

 398            let lib:any = {};

 399            log.info(`loaded ${path}, creating library`);

 400            new Function(code).bind(lib)();

 401            return lib;

 402        } else {

 403            log.warn(`wrong lib format for ${sym}: ${code.slice(0, 20)}..., should be 'this["${sym}"] = ...`);

 404        }

 405    } catch(e) {

 406        log.error(`JSON.parse: ${path}: ${e}`);

 407        return undefined;

 408    }

 409    // add(content.responseText);

 410}
