  14/**

  15 * Creates Documentation on the main panel 

  16 */

  17export class MainDetail extends Layout { 

  18    getComponents(node:Vnode): Vnode {

  19        const lib =  m.route.param('lib');

  20        const field = m.route.param('field');

  21        node.attrs.route = undefined;

  22

  23        let result = DocSets.nodeCount<=0? '...' :  (

  24            (field==='0' || field==='' || field==='overview')? 

  25                getOverview(DocSets.getNode(`${lib}.overview`, lib)) : itemDoc(DocSets.getNode(field, lib))

  26        ); 

  27        return m('.hsdocs_main', [result || `no content for lib ${lib} and field ${field}`]); 

  28    }

  29}

  31/**

  32 * Checks if the project overview is being requested and returns the overview, 

  33 * or `undefined` if not available

  34 * @param mdl the module name to check

  35 * @return Vnode containing the overview file, or `undefined`

  36 */

  37function getOverview(node:DocsNode):Vnode {

  38    return node? m('div.hsdocs_head', commentLong(node.getSignatures()? node.getSignatures()[0] : node)) : 'overview unavailable';

  39}

  41/**

  42 * Creates documentation for standard items in the main panel

  43 * @param mdl the module to document on the main panel

  44 */

  45function itemDoc(node:DocsNode) {

  46    // log.info(`get itemDoc '${node? node.fullPath : 'node undefined'}'`);

  47    return !node? 'no item' : [

  48        node.title(), 

  49        m('div.hsdocs_head', node.commentLong()),

  50        node.members()

  51    ];

  52}
