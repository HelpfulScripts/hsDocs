  15/**

  16 * Creates Documentation on the main panel 

  17 */

  18export class MainDetail extends Layout { 

  19    getComponents(node:Vnode): Vnode {

  20        const lib =  m.route.param('lib');

  21        const field = m.route.param('field');

  22        node.attrs.route = undefined;

  23

  24        let result = getOverview(lib, field) || itemDoc(DocSets.get(lib, field) || ''); 

  25        return m('.hs-main-detail', [result]); 

  26    }

  27}

  29/**

  30 * Checks if the project overview is being requested and returns the overview, 

  31 * or `undefined` if not available

  32 * @param mdl the module name to check

  33 * @return Vnode containing the overview file, or `undefined`

  34 */

  35function getOverview(lib:string, mdl:string):Vnode {

  36    if (mdl === '0' || mdl === '') {  //show module overview

  37        mdl = DocSets.get(lib, `${lib}.overview`);

  38        if (mdl) { // if project has an overview:

  39            return overviewDoc(mdl); 

  40        }

  41    }

  42    return undefined;

  43}

  45/**

  46 * Creates documentation for standard items in the main panel

  47 * @param mdl the module to document on the main panel

  48 */

  49function itemDoc(mdl:any) {

  50    const sig = mdl.signatures? mdl.signatures[0] : mdl;

  51    return m('.hs-item-doc', [

  52        title(mdl, sig),

  53        commentLong(sig),

  54        members(mdl, sig)

  55    ]);

  56}

  58/**

  59 * Creates documentation for the project overview in the main panel

  60 * @param mdl the module to document on the main panel

  61 */

  62function overviewDoc(mdl:any) {

  63    const sig = mdl.signatures? mdl.signatures[0] : mdl;

  64    return m('.hs-item-doc', [

  65        commentLong(sig),

  66    ]);

  67}

  69/**

  70 * renders the title of the main panel.

  71 * ![title image](data/title.png)

  72 * @param mdl the module to document 

  73 * @param sig a signature of the module, or the the module itself

  74 */

  75function title(mdl:any, sig:any): Vnode { 

  76    return m('.hs-item-title', {id: makeID('title', mdl)}, itemDescriptor(mdl, sig)); 

  77}

  79function members(mdl:any, sig:any): Vnode {

  80    if (mdl.groups) {

  81        return m('.hs-item-members', [

  82            ...mdl.groups.map((g:any) => member(g, mdl.lib, true, true)),   // static public

  83            ...mdl.groups.map((g:any) => member(g, mdl.lib, true, false)),  // static protected and private

  84            ...mdl.groups.map((g:any) => member(g, mdl.lib, false, true)),  // instance public

  85            ...mdl.groups.map((g:any) => member(g, mdl.lib, false, false))  // instance protected and private

  86        ]);

  87    } else if (mdl.parameters) {

  88        return m('.hs-item-members', parameter(mdl.parameters, mdl.lib));

  89    } else if (mdl.signatures) {

  90        return m('.hs-item-members', mdl.signatures.map((s:any) => parameter(s.parameters, mdl.lib)));

  91    } else if (mdl.type && mdl.type.declaration) {

  92        mdl.type.declaration.lib = mdl.lib;

  93        return members(mdl.type.declaration, sig);

  94    } else {

  95        return m('.hs-item-members');

  96    }

  97}

  99function parameter(g:any[], lib:string): Vnode {

 100    let content = g.map((c:any) => m('.hs-item-parameter', {id:makeID('parameter', c)}, itemChild(c)));

 101    content.unshift(m('.hs-item-member-title', {id:'parameters'}, m('span', 'Parameters')));

 102    return m('.hs-item-member', content);

 103}

 105function member(group:any, lib:string, statc:boolean, publc: boolean): Vnode {

 106    const resolve           = ((c:number) => DocSets.get(lib, c));

 107    const directChildren    = ((mdl:any) => !mdl['inheritedFrom']);

 108    const inheritedChildren = ((mdl:any) =>  mdl['inheritedFrom']);

 109    const groupMap = {

 110        'External modules': '.hs-item-external-module',

 111        'Constructors':     '.hs-item-constructor',

 112        'Classes':          '.hs-item-class',          

 113        'Interfaces':       '.hs-item-interface',          

 114        'Functions':        '.hs-item-function',          

 115        'Methods':          '.hs-item-method',          

 116        'Variables':        '.hs-item-variable',

 117        'Object literals':  '.hs-item-object-literal',

 118        'Properties':       '.hs-item-property',

 119        'Enumerations':     '.hs-item-enumration',

 120        'Type aliases':     '.hs-item-alias',          

 121        'Accessors':        '.hs-item-accessors'        

 122    };

 123    const fn = groupMap[group.title] || '.hs-item-unknown-member';

 124    const isPublic = (flags:any) => flags.isPublic || (flags.isExported && !flags.isPrivate && !flags.isProtected);

 125

 126    const content = group.children

 127        .map(resolve)

 128        .filter(directChildren)

 129        .filter((mdl:any) => statc? mdl.flags.isStatic : !mdl.flags.isStatic)

 130        .filter((mdl:any) => publc? isPublic(mdl.flags) : !isPublic(mdl.flags))

 131        .map((mdl:any) => m(fn, {id:makeID(group.title, mdl)}, itemChild(mdl)));

 132    const inherited = group.children

 133        .map(resolve)

 134        .filter(inheritedChildren)

 135        .filter((mdl:any) => statc? mdl.flags.isStatic : !mdl.flags.isStatic)

 136        .filter((mdl:any) => publc? mdl.flags.isPublic : !mdl.flags.isPublic)

 137        .map((mdl:any) => m(`.hs-item-inherited ${fn}`, {id:makeID(group.title, mdl)}, itemChild(mdl)));

 138

 139    const publStr = publc?'Public':'Protected or Private';

 140    const statStr = statc?'Static':''; 

 141    if (inherited.length>0) {

 142        inherited.unshift(m('.hs-item-inherited .hs-item-member-title', m('span', `${publStr} ${statStr} Inherited ${group.title}`)));

 143    }

 144    if (content.length>0) {

 145        content.unshift(m('.hs-item-member-title', {id:group.title.toLowerCase()}, m('span', `${publStr} ${statStr} ${group.title}`)));

 146    }

 147    return m(`.hs-item-member ${statc?'.hs-item-static':''} ${publc?'.hs-item-public':''}`, content.concat(inherited));

 148}

 150/**

 151 * prints the descriptor line for a specified signature.

 152 * @param mdl 

 153 * @param sig 

 154 */

 155function itemDescriptor(mdl:any, sig:any):Vnode {

 156    try { return m('.hs-item-desc', [ 

 157            flags(mdl),

 158            kindString(mdl),

 159            itemName(mdl, mdl),

 160            signature(sig, mdl),

 161            type(sig,  mdl.lib),

 162            extensionOf(mdl),

 163            implementationOf(mdl),

 164            inheritedFrom(mdl),

 165            sourceLink(mdl),

 166            extendedBy(mdl),

 167            implementedBy(mdl)

 168        ]);

 169    }

 170    catch(e) { console.log(e); console.log(mdl); }

 171}

 173function itemChild(mdl:any, sig=mdl): Vnode[] {

 174    return mdl.signatures? 

 175        mdl.signatures.map((s:any) => m('.hs-item-child-signature',[itemDescriptor(mdl, s), comment(s, true)])) : 

 176        [itemDescriptor(mdl, sig), comment(sig,true)];

 177}
