  17/**

  18 * Creates Documentation on the main panel 

  19 */

  20export class MainDetail extends Widget { 

  21    view(node:Vnode) {

  22        const lib = node.attrs.lib;

  23        const field = node.attrs.field;

  24        let result = DocSets.nodeCount<=0? '...' :  (

  25            (field==='0' || field==='' || field==='overview')? 

  26                getOverview(DocSets.getNode(`${lib}.overview`, lib)) 

  27              : itemDoc(DocSets.getNode(field, lib))

  28        ); 

  29        return m('.hsdocs_detail', node.attrs, [result || `no content for lib ${lib} and field ${field}`]); 

  30    }

  31}

  33/**

  34 * Checks if the project overview is being requested and returns the overview, 

  35 * or `undefined` if not available

  36 * @param mdl the module name to check

  37 * @return Vnode containing the overview file, or `undefined`

  38 */

  39function getOverview(node:DocsNode):m.Child {

  40    return node? m('div.hsdocs_head_comment', (node.getSignatures()? node.getSignatures()[0] : node).commentLong()) : 'overview unavailable';

  41}

  42

  43/**

  44 * Creates documentation for standard items in the main panel

  45 * @param mdl the module to document on the main panel

  46 */

  47function itemDoc(node:DocsNode) {

  48    return !node? 'no item' : [

  49        node.title(), 

  50        node.getSignatures()? undefined : m('div.hsdocs_head_comment', node.commentLong()),

  51        node.members()

  52    ];

  53}
