DocSets.ts

   1/**

   2 * DocSets.ts. Loads the doc.json files to process and display as documentation.

   3 * Processing occurs in these steps:

   4 * 1. Load the index.json file that contains a list of doc.json files to load, one for each library to show

   5 * 2. Load each doc.json file, which describes a library

   6 * 3. Call DocSets.add to add the library name to the registry and create an index of entries for the library

   7 */

   8

  24function matchAll(str:string, re:RegExp): string[] {

  25    const result:string[] = [];

  26    let partial:string[];

  27    while ((partial = re.exec(str)) !== null) {

  28        result.push(partial[1]);

  29        log.debug(()=>log.inspect(partial));

  30    }

  31    return result;

  32}

  34/**

  35 * DocSets object. Keeps a list of registered docsets and 

  36 * provides access to elements of each docset.

  37 */

  38export class DocSets { 

  39    /** Contains references to the docsets and all elements per docset, accessible per ID. */

  40    // private static gList:List = {index:{}};

  41

  42    private static libs:string[] = [];

  43    private static docs:string[] = [];

  44

  45    /** an optional title for the set of DocSets loaded, as specified in the `index.json` file. */

  46    private static gTitle: string = '';

  47    private static nodeList = <{[id:string]: DocsNode}>{};

  48    

  49    public static nodeCount = 0;

  50    

  51    public static addNode(mdl:json, node:DocsNode) {

  52        const id = mdl.id;

  53        const lib =  mdl.lib;

  54        DocSets.nodeList[`${lib}.${id}`] = node;

  55        DocSets.nodeList[mdl.fullPath] = node;

  56// if (mdl.fullPath.indexOf('hsDocs.DocSets')>=0) { log.info(`added node ${id}: '${mdl.fullPath}'`); }

  57        DocSets.nodeCount++;

  58    }

  60

  61    /**

  62     * Adds the docset in `content` to the `gList` at the position of `file` in `DocSets.docs`.

  63     * @param content the docSet content to add. 

  64     * @param file the file name from which the content was read. 

  65     */

  66    public static addDocSet(content:json, file:string):void {

  67        const lib = content.name;

  68        const i = DocSets.docs.indexOf(file);

  69        DocSets.libs[i] = lib;

  70        const root = DocsNode.traverse(content, lib);

  71        log.debug(`traversed '${root.fullPath}'`);

  72    }

  73    

  74    public static getNode(id:string|number, lib:string):DocsNode {

  75        const key = ((typeof id === 'number') || (parseInt(''+id, 10)>=0))? `${lib}.${id}` : id;

  76        log.debug(() => `getNode id=${id}, lib=${lib} -> ${key}, typeof=${typeof id}, parseInt=${parseInt(''+id, 10)}`);

  77        if (DocSets.nodeList.length && !DocSets.nodeList[key]) { 

  78            log.warn(`did not find node for key '${key}' (id=${id}, lib=${lib})`); 

  79            log.warn(new Error().stack);

  80        }

  81        return DocSets.nodeList[key]; 

  82    }

  84    /**

  85     * returns the specified documentation element.

  86     * When called without parameters, a `string[lib]` of available docSets is returned.

  87     * When called with only `lib` specified, the corresponding docSet overview is returned.

  88     * @param lib specifies the docset to scan. 

  89     * @param id specifies the element within the docSet, either by its id number, or by its fully qualified path, 

  90     * e.g. 'hsDocs.DocSets.DocSets.add' 

  91     */

  92    public static getLibs():string[]{ 

  93        return DocSets.libs; 

  94    }

  96    /**

  97     * loads an index set and the docsets it contains from directory `dir`.

  98     * @param file the optional directory to read from. If unspecified, 

  99     * read from the index file specified by {@link DocSets.FILE `FILE`}.

 100     * @return a promise to load the index set

 101     */

 102    public static async loadList(file?:string):Promise {

 103        /** load an `index.json` file that contains references to the DocSets to load. */

 104        async function getIndexFile(dir:string, url:string):Promise {

 105            if (!url) { return false; }

 106            try {

 107                const result = await m.request({ method: "GET", url: file});

 108                DocSets.gTitle = result.title;

 109                DocSets.docs = result.docs.map((doc:string) => doc.indexOf(':')>0? doc : dir+doc);

 110                log.debug(`found index file ${url} with ${DocSets.docs.length} library references`);

 111                return true;

 112            } catch(e) {

 113                return false;

 114            }

 115        }

 116        /** 

 117         * get all `jsons` in a directory specified by `url`. 

 118         * If the directory contains an `index.json`, calls `getIndexFile` on that file's url.

 119         */

 120        async function getDirJSONs(url:string):Promise {

 121            let result = false;

 122            await m.request({ method: "GET", url: url, 

 123                // catch extract to avoid attempt to deserialize

 124                extract: async (xhr:any, options:any) => { 

 125                    const jsons = matchAll(xhr.responseText, /"(\S*?.json)">/g);  // all *.json references

 126                    // check if index.jon exists:

 127                    const index = jsons.filter(m => m==='index.json')[0];

 128                    // const jsons = matches.filter(m => !m.startsWith('index'));

 129

 130                    if (index) { 

 131                        result = await getIndexFile(dir, index); 

 132                    } else if (jsons.length) {

 133                        log.debug(`found dir list in ${url}`);

 134                        DocSets.gTitle = '';

 135                        DocSets.docs = jsons;

 136                        result = true;

 137                    }

 138                }

 139            });

 140            return result;

 141        }

 142        let i = file.lastIndexOf('/');

 143        const dir = file.substring(0,i+1);

 144        

 145        let found = false;

 146        if (!found) { found = await getIndexFile(dir, file); }

 147        if (!found) { found = await getDirJSONs(dir); }

 148        if (!found) { found = await getDirJSONs(DOCDIR); }

 149        log.debug(()=>`found ${DocSets.docs.length} dos sets: ${log.inspect(DocSets.docs, {depth:5})}`);

 150        await Promise.all(DocSets.docs.map(async (f:string) => await loadDocSet(f))).catch(log.error);        

 151        log.debug(`found ${DocSets.nodeCount} DocNodes`);

 152        m.redraw();

 153    }

 154}

 156/**

 157 * Loads a docset specified by file from the directory `dir`. 

 158 * Once received, the docset is registered in `modules` via the `add` method.

 159 * @param dir the directory to read from

 160 * @param file the `json` file to load as docset

 161 */

 162async function loadDocSet(file:string):Promise {

 163    log.debug(()=>`loading ${file}`);

 164    const r:json = await m.request({ method: "GET", url: file });

 165    DocSets.addDocSet(r, file);

 166}
