   1/**

   2 * DocSets.ts. Loads the doc.json files to process and display as documentation.

   3 * Processing occurs in these steps:

   4 * 1. Load the index.json file that contains a list of doc.json files to load, one for each library to show

   5 * 2. Load each doc.json file, which describes a library

   6 * 3. Call DocSets.add to add the library name to the registry and create an index of entries for the library

   7 */

   8

   9/** */

  10import { m }            from '../node_modules/hslayout/index.js';

  11import { log as gLog }  from '../node_modules/hsutil/index'; const log = gLog('DocsSets');

  12

  13const DOCDIR:string = './data/';

  14

  15/**

  16 * the default location for the index .json files, relative to the web page:

  17 * `'./data/index.json'`

  18 */

  19const FILE:string = './data/index.json';

  20

  21function matchAll(str:string, re:RegExp): string[] {

  22    const result:string[] = [];

  23    let partial:string[];

  24    while ((partial = re.exec(str)) !== null) {

  25        result.push(partial[1]);

  26        log.debug(log.inspect(partial));

  27    }

  28    return result;

  29}

  31/**

  32 * DocSets object. Keeps a list of registered docsets and 

  33 * provides access to elements of each docset.

  34 */

  35export class DocSets { 

  36    /** Contains references to the docsets and all elements per docset, accessible per ID. */

  37    private static gList = <{set:string[], index:{}, docs:string[]}>{set:[], index:{}, docs:[]};

  38    private static gTitle: string;

  39

  40    /**

  41     * Adds the docset in `content` to the `gList` at the position of `file` in `DocSets.gList.docs`.

  42     * @param content the docSet content to add. 

  43     * @param file the file name from which the content was read. 

  44     */

  45    public static add(content:any, file:string) {

  46        const lib = content.name;

  47        const i = DocSets.gList.docs.indexOf(file);

  48        DocSets.gList.set[i] = lib;

  49        DocSets.gList.index[lib] = {};

  50        recursiveIndex(content, DocSets.gList.index[lib], lib);

  51    }

  52

  53    /**

  54     * loads an index set and the docsets it contains from driectory `dir`.

  55     * @param file the optional directory to read from. If unspecified, 

  56     * read from the index file specified by {@link DocSets.FILE `FILE`}.

  57     * @return a promise to load the index set

  58     */

  59    public static async loadList(file?:string):Promise {

  60        async function getIndexFile(url:string):Promise {

  61            if (!url) { return false; }

  62            try {

  63                const result = await m.request({ method: "GET", url: file});

  64                DocSets.gTitle = result.title;

  65                DocSets.gList.docs = result.docs;

  66                log.info(`found index file ${url}`);

  67                return true;

  68            } catch(e) {

  69                return false;

  70            }

  71        }

  72        async function getDirJSONs(url:string):Promise {

  73            let result = false;

  74            await m.request({ method: "GET", url: url, 

  75                // catch extract to avoid attempt to deserialize

  76                extract: async (xhr:any, options:any) => { 

  77                    const matches = matchAll(xhr.responseText, /"(\S*?.json)">/g);  // all *.json references

  78                    // check if index.jon exists:

  79                    const index = matches.filter(m => m==='index.json')[0];

  80                    const jsons = matches.filter(m => !m.startsWith('index'));

  81

  82                    if (index) { 

  83                        result = await getIndexFile(index); 

  84                    } else if (jsons.length) {

  85                        log.info(`found dir list ${url}`);

  86                        DocSets.gTitle = '';

  87                        DocSets.gList.docs = jsons;

  88                        result = true;

  89                    }

  90                }

  91            });

  92            return result;

  93        }

  94        file = file;   

  95        let i = file.lastIndexOf('/');

  96        const dir = file.substring(0,i+1);

  97        

  98        console.log('loading docs dir');

  99        let found = false;

 100        if (!found) { found = await getIndexFile(file); }

 101        if (!found) { found = await getDirJSONs(dir); }

 102        if (!found) { found = await getDirJSONs(DOCDIR); }

 103        log.info(`found ${DocSets.gList.docs.length} dos sets: ${log.inspect(DocSets.gList.docs, 5)}`);

 104        await Promise.all(DocSets.gList.docs.map(async (f:string) => await loadDocSet(dir, f)));            

 105    }

 106

 107    /**

 108     * returns the specified documentation element.

 109     * When called without parameters, a `string[lib]` of available docSets is returned.

 110     * When called with only `lib` specified, the corresponding docSet overview is returned.

 111     * @param lib specifies the docset to scan. 

 112     * @param id specifies the element within the docSet, either by its id number, or by its fully qualified path, 

 113     * e.g. 'hsDocs.DocSets.DocSets.add' 

 114     */

 115    public static get(lib?:string, id:number|string=0) { 

 116        return (lib && DocSets.gList.index[lib])?

 117            DocSets.gList.index[lib][id+'']

 118          : DocSets.gList.set; 

 119    }

 120}

 121

 122/**

 123 * Loads a docset specified by file from the directory `dir`. 

 124 * Once received, the docset is registered in `modules` via the `add` method.

 125 * @param dir the directory to read from

 126 * @param file the `json` file to load as docset

 127 */

 128function loadDocSet(dir:string, file:string):Promise {

 129    return m.request({ method: "GET", url: dir+file })

 130        .then((r:any) => DocSets.add(r, file))

 131        .catch(console.log);

 132}

 133

 134/**

 135 * recurses through the docset and registers all `children` entries of an entry by id,

 136 * starting with the root entry.

 137 * @param content the docset object literal to traverse

 138 * @param index the index in which to register the entries

 139 * @param lib the docset name, used for name validation

 140 */

 141function recursiveIndex(content:any, index:any, lib:string, path='') {

 142    function getNewPath(content:any) {

 143        content.name = content.name.replace(/["'](.+)["']|(.+)/g, "$1$2");  // remove quotes 

 144        // const elName  = content.name.match(/([^\/]+)$/)[1];         // name = part after last /

 145        const elName  = content.name.replace(/\//g, '.');         // "a/b" => "a.b" /

 146        content.name = elName;

 147        return content.fullPath = (path==='')? elName : `${path}.${elName}`;

 148    }

 149

 150    function markIfModule(content:any) {

 151        if (content.comment && content.comment.tags) {

 152            content.comment.tags.forEach((tag:any) => {

 153                if (tag.tag === 'module') {

 154                    content.innerModule = tag.text.trim();

 155                }

 156            });

 157        }

 158    }

 159    content.lib = lib;

 160    if (typeof content === 'object' && content.name) {

 161        const newPath = getNewPath(content);

 162

 163        markIfModule(content);

 164

 165        index[content.id+''] = content;

 166        if (newPath.length>0) { index[newPath] = content; }

 167

 168        if (content.children) {

 169            content.children.map((c:any) => recursiveIndex(c, index, lib, newPath));

 170        }

 171        if (content.signatures) {

 172            content.signatures.map((c:any) => recursiveIndex(c, index, lib, newPath));

 173        }

 174        if (content.parameters) {

 175            content.parameters.map((c:any) => recursiveIndex(c, index, lib, newPath));

 176        }

 177        if (content.type && content.type.declaration && content.type.declaration.children) {

 178            content.type.declaration.children.map((c:any) => recursiveIndex(c, index, lib, newPath));

 179        }

 180    }
