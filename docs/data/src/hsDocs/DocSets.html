src/DocSets.ts

   1/**

   2 * DocSets.ts. Loads the doc.json files to process and display as documentation.

   3 * Processing occurs in these steps:

   4 * 1. Load the index.json file that contains a list of doc.json files to load, one for each library to show

   5 * 2. Load each doc.json file, which describes a library

   6 * 3. Call DocSets.add to add the library name to the registry and create an index of entries for the library

   7 */

   8

   9/** */

  10import { m }    from 'hslayout';

  11

  12/**

  13 * the default location for the index .json files, relative to the web page:

  14 * `'./data/index.json'`

  15 */

  16const FILE:string = './data/index.json';

  17

  18

  19/**

  20 * DocSets object. Keeps a list of registered docsets and 

  21 * provides access to elements of each docset.

  22 */

  23export class DocSets { 

  24    /** Contains references to the docsets and all elements per docset, accessible per ID. */

  25    private static gList = <{set:string[], index:{}, docs:[string]}>{set:[], index:{}, docs:[]};

  26    private static gTitle: string;

  27

  28    /**

  29     * Adds the docset in `content` to the `gList` at the position of `file` in `DocSets.gList.docs`.

  30     * @param content the docSet content to add. 

  31     * @param file the file name from which the content was read. 

  32     */

  33    public static add(content:any, file:string) {

  34        const lib = content.name;

  35        const i = DocSets.gList.docs.indexOf(file);

  36        DocSets.gList.set[i] = lib;

  37        DocSets.gList.index[lib] = {};

  38        recursiveIndex(content, DocSets.gList.index[lib], lib);

  39    }

  40

  41    /**

  42     * loads an index set and the docsets it contains from driectory `dir`.

  43     * @param file the optional directory to read from. If unspecified, 

  44     * read from the index file specified by {@link DocSets.FILE `FILE`}.

  45     * @return a promise to load the index set

  46     */

  47    public static loadList(file?:string):Promise {

  48        file = file;   

  49        let i = file.lastIndexOf('/');

  50        const dir = file.substring(0,i+1);

  51        return m.request({ method: "GET", url: file || FILE})

  52        .then((result:any) => {

  53            DocSets.gTitle = result.title;

  54            DocSets.gList.docs = result.docs;

  55            return Promise.all(result.docs.map((f:string) => loadDocSet(dir, f)));            

  56        })

  57        .catch(console.log);

  58    }

  59

  60    /**

  61     * returns the specified documentation element.

  62     * When called without parameters, a `string[lib]` of available docSets is returned.

  63     * When called with only `lib` specified, the corresponding docSet overview is returned.

  64     * @param lib specifies the docset to scan. 

  65     * @param id specifies the element within the docSet, either by its id number, or by its fully qualified path, 

  66     * e.g. 'hsDocs.DocSets.DocSets.add' 

  67     */

  68    public static get(lib?:string, id:number|string=0) { 

  69        return (lib && DocSets.gList.index[lib])?

  70            DocSets.gList.index[lib][id+'']

  71          : DocSets.gList.set; 

  72    }

  73}

  74

  75/**

  76 * Loads a docset specified by file from the directory `dir`. 

  77 * Once received, the docset is registered in `modules` via the `add` method.

  78 * @param dir the directory to read from

  79 * @param file the `json` file to load as docset

  80 */

  81function loadDocSet(dir:string, file:string):Promise {

  82    return m.request({ method: "GET", url: dir+file })

  83        .then((r:any) => DocSets.add(r, file))

  84        .catch(console.log);

  85}

  86

  87/**

  88 * recurses through the docset and registers all `children` entries of an entry by id,

  89 * starting with the root entry.

  90 * @param content the docset object literal to traverse

  91 * @param index the index in which to register the entries

  92 * @param lib the docset name, used for name validation

  93 */

  94function recursiveIndex(content:any, index:any, lib:string, path='') {

  95    function getNewPath(content:any) {

  96        content.name = content.name.replace(/["'](.+)["']|(.+)/g, "$1$2");  // remove quotes 

  97        const elName  = content.name.match(/([^\/]+)$/)[1];         // name = part after last /

  98        content.name = elName;

  99        return content.fullPath = (path==='')? elName : `${path}.${elName}`;

 100    }

 101

 102    function markIfModule(content:any) {

 103        if (content.comment && content.comment.tags) {

 104            content.comment.tags.forEach((tag:any) => {

 105                if (tag.tag === 'module') {

 106                    content.innerModule = tag.text.trim();

 107                }

 108            });

 109        }

 110    }

 111    content.lib = lib;

 112    if (typeof content === 'object' && content.name) {

 113        const newPath = getNewPath(content);

 114

 115        markIfModule(content);

 116

 117        index[content.id+''] = content;

 118        if (newPath.length>0) { index[newPath] = content; }

 119

 120        if (content.children) {

 121            content.children.map((c:any) => recursiveIndex(c, index, lib, newPath));

 122        }

 123        if (content.signatures) {

 124            content.signatures.map((c:any) => recursiveIndex(c, index, lib, newPath));

 125        }

 126        if (content.parameters) {

 127            content.parameters.map((c:any) => recursiveIndex(c, index, lib, newPath));

 128        }

 129        if (content.type && content.type.declaration && content.type.declaration.children) {

 130            content.type.declaration.children.map((c:any) => recursiveIndex(c, index, lib, newPath));

 131        }

 132    }

 133}
