  77class DocsType {

  78    static makeType(mdlType:DocsGenericType) {

  79        switch (mdlType.type) {

  80            case 'intrinsic': return new DocsIntrinsicType(mdlType);

  81            case 'array': return new DocsArrayType(mdlType);

  82            case 'reflection': return new DocsReflectionType(mdlType);

  83            case 'stringLiteral': return new DocsStringLiteralType(mdlType);

  84            case 'reference': return new DocsReferenceType(mdlType);

  85            case 'tuple': return new DocsTupleType(mdlType);

  86            case 'union': return new DocsUnionType(mdlType);

  87            case 'unknown': return new DocsUnknownType(mdlType);

  88            default: log.warn(`unknown type '${mdlType.type}' in makeType`);

  89

  90        }

  91    }

  92    type: string;

  93    constructor(mdlType:DocsGenericType) {

  94        this.type = mdlType.type;

  95    }

 167export class DocsNode {

 168    /** fields to expculde from the test */

 169    static meta = ['FLAGS', 'TYPE'];

 170    static count = 0;

 171    static makeNode(mdl:any) {

 172        const node = groupKindTitle[mdl.kind];

 173        if (!node) { log.warn(`unknown kind ${mdl.kind}`); }

 174        else { return new node(mdl); }

 175    }

 176

 177    static traverse(mdl:any) {

 178        DocsNode.count++;

 179        const node = DocsNode.makeNode(mdl);

 180        if (mdl.children) {

 181            mdl.children.forEach((c:any) => DocsNode.count<5000? DocsNode.traverse(c) : '');

 182        }

 183    }

 184

 185    id:number   = 0;  // defined = required; null = optional

 186    kind:number = 0;

 187    kindString: string = '';

 188    TYPE = [];

 189    type: DocsType = null;

 190

 191    constructor(mdl:any) {

 192        this.id = mdl.id;

 193        this.kindString = mdl.kindString || 'root';

 194    }

 195

 196    set(mdl:any, fields = Object.getOwnPropertyNames(this)) {

 197        fields.forEach(f => { // expected fields

 198            if (mdl[f]!==undefined) { 

 199                if (f==='type') {

 200                    if (this.TYPE && this.TYPE.indexOf(mdl[f].type)>=0) {

 201                        this.type = DocsType.makeType(mdl[f]);

 202                    } else {

 203                        log.warn(`'${this.kindString}' #${mdl.id}: unexpected type '${mdl[f].type}'`);

 204                    }

 205                } else {

 206                    this[f] = mdl[f]; 

 207                }

 208            } else if (!ifOptional(this[f]) && DocsNode.meta.indexOf(f)<0) {

 209                log.warn(`'${this.kindString}' #${mdl.id}: field '${f}' should be optional`);

 210            }

 211        });

 212        fields = Object.getOwnPropertyNames(mdl);

 213        fields.forEach(f => { // unexpected fields

 214            if (!ifExists(this[f])) {

 215                log.warn(`'${this.kindString}' #${this.id}: found extra field '${f}'='${log.inspect(mdl[f],2)}'`);

 216            }

 217        });

 218    }

 221class DocsFlagged extends DocsNode {

 222    FLAGS = {};  // flags listed here are allowed if false and required if true

 223    flags: flags.Flags = {};

 224

 225    constructor(mdl:any) {

 226        super(mdl);

 227    }

 228

 229    setFlags(mdlFlags:any) {

 230        const staticFlags:Flags = this.FLAGS;

 231        Object.keys(staticFlags).forEach(f => { // expected flags

 232            if (mdlFlags[f]!==undefined) { this.flags[f] = mdlFlags[f]; }

 233            else if (staticFlags[f]===true) {

 234                log.warn(`'${this.kindString}' #${this.id}: flag '${f}' should be optional`);

 235            }

 236        });

 237        const flags = Object.getOwnPropertyNames(mdlFlags);

 238        flags.forEach(f => { // unexpected flags

 239            if (staticFlags[f]===undefined) {

 240                log.warn(`'${this.kindString}' #${this.id}: found extra flag '${f}'='${log.inspect(mdlFlags[f],1)}'`);

 241            }

 242        });

 243    }
