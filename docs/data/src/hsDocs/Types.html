  31export class DocsType implements DocsReferenceIdType {

  32    static makeType(mdlType:DocsGenericType, node:DocsNode) {

  33        switch (mdlType.type) {

  34            case 'intrinsic': return new DocsIntrinsicType(mdlType, node);

  35            case 'array': return new DocsArrayType(mdlType, node);

  36            case 'reflection': return new DocsReflectionType(mdlType, node);

  37            case 'stringLiteral': return new DocsStringLiteralType(mdlType, node);

  38            case 'reference': return new DocsReferenceType(mdlType, node);

  39            case 'tuple': return new DocsTupleType(mdlType, node);

  40            case 'union': return new DocsUnionType(mdlType, node);

  41            case 'typeParameter': return new DocsTypeParameterType(mdlType, node);

  42            case 'typeOperator': return new DocsTypeOperatorType(mdlType, node);

  43            case 'unknown': return new DocsUnknownType(mdlType, node);

  44            default: log.warn(`unknown type '${mdlType.type}' in makeType`);

  45        }

  46    }

  47    type: string;

  48    name: string;

  49    node: DocsNode;

  50    id:   number;

  51    declaration: DocsNode;

  52    constructor(mdlType:DocsGenericType, node:DocsNode) {

  53        this.type = mdlType.type;

  54        this.id   = mdlType.id;

  55        this.name = mdlType.name;

  56        this.node = node;

  57    }

  58

  59    mType() { return m('span.hsdocs_type_unknown', 'default type'); }

  60}

  62class DocsIntrinsicType extends DocsType {

  65        super(mdlType, node);

  66        this.name = mdlType.name;

  67    }

  68    mType() {

  69        return m('span.hsdocs_type_intrinsic', this.id? libLink(this.node.lib, this.id, this.name) : this.name); 

  70    }

  73class DocsReferenceType extends DocsType {

  74    typeArguments: DocsType[];

  75    constructor(mdlType:DocsGenericType, node:DocsNode) {

  76        super(mdlType, node);

  77        this.typeArguments = !mdlType.typeArguments? [] : mdlType.typeArguments.map((t:DocsNamedType) => DocsType.makeType(t, node));

  78    }

  79

  80    mType() {

  81        let refRes = this.id? libLink(this.node.lib, this.id, this.name) : this.name;

  82        if (this.typeArguments.length) { refRes += '<'+ this.typeArguments.map(t => t.mType()).join(', ') + '>'; }

  83        // return m('span.hsdocs_type_reference', refRes);        

  84        return refRes;        

  85    }

 175class DocsTupleType extends DocsType {

 176    elements = [];

 177    constructor(mdlType:DocsGenericType, node:DocsNode) {

 178        super(mdlType, node);

 179        mdlType.elements.forEach((e:DocsType) => this.elements.push(DocsType.makeType(e, node)));

 180    }

 181    mType() {

 182        return m('span.hsdocs_type_tuple', [

 183            '[ ',

 184            ...this.elements.map((e:any, i:number) => [i>0?', ':undefined, e.mType()]),

 185            ' ]'

 186        ]);

 187    }
