  37export class DocsType implements DocsReferenceIdType {

  38    static makeType(mdlType:DocsGenericType, node:DocsNode) {

  39        switch (mdlType.type) {

  40            case 'intrinsic': return new DocsIntrinsicType(mdlType, node);

  41            case 'array': return new DocsArrayType(mdlType, node);

  42            case 'reflection': return new DocsReflectionType(mdlType, node);

  43            case 'stringLiteral': return new DocsStringLiteralType(mdlType, node);

  44            case 'reference': return new DocsReferenceType(mdlType, node);

  45            case 'tuple': return new DocsTupleType(mdlType, node);

  46            case 'union': return new DocsUnionType(mdlType, node);

  47            case 'typeParameter': return new DocsTypeParameterType(mdlType, node);

  48            case 'typeOperator': return new DocsTypeOperatorType(mdlType, node);

  49            case 'query': return new DocsQueryType(mdlType, node);

  50            case 'indexedAccess': return new DocsIndexedAccessType(mdlType, node);

  51            case 'unknown': return new DocsUnknownType(mdlType, node);

  52            default: log.warn(`unknown type '${mdlType.type}' for ${node.fullPath} in makeType`);

  53        }

  54    }

  55    type: string;

  56    name: string;

  57    node: DocsNode;

  58    id:   number;

  59    declaration: DocsNode;

  60    constructor(mdlType:DocsGenericType, node:DocsNode) {

  61        this.type = mdlType.type;

  62        this.id   = mdlType.id;

  63        this.name = mdlType.name;

  64        this.node = node;

  65    }

  66

  67    mType() { return m('span.hsdocs_type_unknown', 'default type'); }

  68}

  73        super(mdlType, node);

  74        this.name = mdlType.name;

  76    mType() {

  77        // return m('span.hsdocs_type_intrinsic', this.id? libLinkByPath(this.node.lib, ''+this.id, this.name) : parent.getName()); 

  78        return m('span.hsdocs_type_intrinsic', this.id? libLinkByPath(this.node.lib, ''+this.id, this.name) : this.name); 

  79    }

  82class DocsReferenceType extends DocsType {

  83    typeArguments: DocsType[];

  84    constructor(mdlType:DocsGenericType, node:DocsNode) {

  85        super(mdlType, node);

  86        this.typeArguments = !mdlType.typeArguments? [] : mdlType.typeArguments.map((t:DocsNamedType) => DocsType.makeType(t, node));

  87    }

  88

  89    mType() {

  90        let refRes = [this.id? libLinkByPath(this.node.lib, ''+this.id, this.name) : this.name];

  91        if (this.typeArguments.length) { 

  92            refRes.push('<');

  93            refRes.push(...this.typeArguments.map(t => t.mType()));

  94            refRes.push('>'); 

  95        }

  96        return m('span.hsdocs_type_reference', refRes);        

  97    }

  98}

 190class DocsTupleType extends DocsType {

 191    elements = [];

 192    constructor(mdlType:DocsGenericType, node:DocsNode) {

 193        super(mdlType, node);

 194        mdlType.elements.forEach((e:DocsType) => this.elements.push(DocsType.makeType(e, node)));

 195    }

 196    mType() {

 197        return m('span.hsdocs_type_tuple', [

 198            '[ ',

 199            ...this.elements.map((e:any, i:number) => [i>0?', ':undefined, e.mType()]),

 200            ' ]'

 201        ]);

 202    }

 205class DocsUnionType extends DocsType {

 214    }
