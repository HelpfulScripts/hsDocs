 147class DocsBase {

 148    id:     number = null;  // null = required; undefined = optional

 149    kind:   number = null;

 150

 151    constructor(node:DocsBase) {

 152        this.set(['id', 'kind'], node);

 153    }

 154

 155    set(fields:string[], node:DocsBase) {

 156        fields.forEach(f => this[f] = node[f] || this[f]);

 157    }

 158    /**

 159     * tests this node against the original data `node` from which it was created.

 160     * @param node 

 161     */

 162    static testMissingOrIllegalFields(template:DocsNode, data:DocsNode) {

 163        let names =  Object.getOwnPropertyNames(data);

 164        names.map(k => {

 165            if (template[k]===undefined) { 

 166                log.warn(`'${template.kindString}' #${data.id}: found extra key '${k}'='${data[k]}'`);

 167            }

 168        });

 169        names = Object.getOwnPropertyNames(template);

 170        names.map(k => {

 171            if (template[k]===null) { // is required, should be optional

 172                log.warn(`'${template.kindString}' #${data.id}: key '${k}' should be optional`);

 173            }

 174        });

 175    }

 564function testFullNode(node:DocNode, name=''):number {

 565    if (name==='') { name = node.name; }

 566    const num =  Object.getOwnPropertyNames(node).map((k):number => {

 567        use('DocNode', k);

 568        switch (k) {

 569            case 'lib':             return 0;  // added by HsDocs

 570            case 'modules':         return 0;  // added by HsDocs

 571            case 'fullPath':        return 0;  // added by HsDocs

 572            case 'id':              return testNumber(k, node.id, node, node.name);

 573            case 'name':            return testString(k, node.name, node,`${name}-${k}`);

 574            case 'kind':            return testNumber(k, node.kind, node,`${name}-${k}`);

 575            case 'kindString':      return testString(k, node.kindString, node,`${name}-${k}`);

 576            case 'flags':           return testFlags(node.flags, node,`${name}-${k}`);

 577            case 'originalName':    return testString(k, node.originalName, node,`${name}-${k}`);

 578            case 'defaultValue':    return testString(k, node.defaultValue, node,`${name}-${k}`);

 579            case 'comment':         return testComment(node.comment, node,`${name}-${k}`);

 580            case 'sources':         return testSources(node.sources, node,`${name}-${k}`);

 581            case 'type':            return testType(node.type, node,`${name}-${k}`);

 582            case 'inheritedFrom':   return testTypeNameID(node.inheritedFrom, node,`${name}-${k}`);

 583            case 'implementationOf':return testTypeNameID(node.implementationOf, node,`${name}-${k}`);

 584            case 'overwrites':      return testTypeNameID(node.overwrites, node,`${name}-${k}`);

 585            case 'implementedTypes':return node.implementedTypes.map(t => testTypeNameID(t, node,`${name}-${k}`)).reduce(accumulate, 0);

 586            case 'implementedBy':   return node.implementedBy.map(t => testTypeNameID(t, node,`${name}-${k}`)).reduce(accumulate, 0);

 587            case 'extendedBy':      return node.extendedBy.map(e => testTypeNameID(e, node,`${name}-${k}`)).reduce(accumulate, 0);

 588            case 'extendedTypes':   return node.extendedTypes.map(t =>testExtendedTypes(t, node,`${name}-${k}`)).reduce(accumulate, 0);

 589            case 'children':        return node.children.map(c => testFullNode(c, `${name}-${c.name}`)).reduce(accumulate, 0);

 590            case 'groups':          return node.groups.map(c => testReference(c, node,`${name}-${k}`)).reduce(accumulate, 0);

 591            case 'signatures':      return node.signatures.map(c => testSignature(c, node, `${name}-${k}`)).reduce(accumulate, 0);

 592            case 'parameters':      return node.parameters.map(c => testFullNode(c, `${name}-${k}`)).reduce(accumulate, 0);
