 175export function testFullNode(node:DocNode, name=''):number {

 176    if (name==='') { name = node.name; }

 177    const num =  Object.getOwnPropertyNames(node).map((k):number => {

 178        use('DocNode', k);

 179        switch (k) {

 180            case 'lib':             return 0;  // added by HsDocs

 181            case 'modules':         return 0;  // added by HsDocs

 182            case 'fullPath':        return 0;  // added by HsDocs

 183            case 'id':              return testNumber(k, node.id, node, node.name);

 184            case 'name':            return testString(k, node.name, node,`${name}-${k}`);

 185            case 'kind':            return testNumber(k, node.kind, node,`${name}-${k}`);

 186            case 'kindString':      return testString(k, node.kindString, node,`${name}-${k}`);

 187            case 'flags':           return testFlags(node.flags, node,`${name}-${k}`);

 188            case 'originalName':    return testString(k, node.originalName, node,`${name}-${k}`);

 189            case 'defaultValue':    return testString(k, node.defaultValue, node,`${name}-${k}`);

 190            case 'comment':         return testComment(node.comment, node,`${name}-${k}`);

 191            case 'sources':         return testSources(node.sources, node,`${name}-${k}`);

 192            case 'type':            return testType(node.type, node,`${name}-${k}`);

 193            case 'inheritedFrom':   return testTypeNameID(node.inheritedFrom, node,`${name}-${k}`);

 194            case 'implementationOf':return testTypeNameID(node.implementationOf, node,`${name}-${k}`);

 195            case 'overwrites':      return testTypeNameID(node.overwrites, node,`${name}-${k}`);

 196            case 'implementedTypes':return node.implementedTypes.map(t => testTypeNameID(t, node,`${name}-${k}`)).reduce(accumulate, 0);

 197            case 'implementedBy':   return node.implementedBy.map(t => testTypeNameID(t, node,`${name}-${k}`)).reduce(accumulate, 0);

 198            case 'extendedBy':      return node.extendedBy.map(e => testTypeNameID(e, node,`${name}-${k}`)).reduce(accumulate, 0);

 199            case 'extendedTypes':   return node.extendedTypes.map(t =>testExtendedTypes(t, node,`${name}-${k}`)).reduce(accumulate, 0);

 200            case 'children':        return node.children.map(c => testFullNode(c, `${name}-${c.name}`)).reduce(accumulate, 0);

 201            case 'groups':          return node.groups.map(c => testReference(c, node,`${name}-${k}`)).reduce(accumulate, 0);

 202            case 'signatures':      return node.signatures.map(c => testSignature(c, node, `${name}-${k}`)).reduce(accumulate, 0);

 203            case 'parameters':      return node.parameters.map(c => testFullNode(c, `${name}-${k}`)).reduce(accumulate, 0);

 204            case 'indexSignature':  return node.indexSignature.map(is => testFullNode(is, `${name}-${k}`)).reduce(accumulate, 0);

 205            case 'getSignature':    return node.getSignature.map(is => testFullNode(is, `${name}-${k}`)).reduce(accumulate, 0);

 206            case 'typeParameter':   return node.typeParameter.map(is => testBaseNode(is, node, `${name}-${k}`)).reduce(accumulate, 0);

 207            default: log.warn(`testFullNode found unknown key '${k}' for path '${name}' ${node['lib']?'in lib '+node['lib'] : ''}`);

 208        }

 209        return -1;

 210    }).reduce(accumulate, 0);

 211    log.debug(`${num} tests in node ${node.name}`);

 212    return num;

 213}

 400        use('DocReference', k);

 401        switch (k) {

 402            case 'title':       return testString(k, test.title, node,`${name}-${k}`);

 403            case 'kind':        return testNumber(k, test.kind, node,`${name}-${k}`);

 404            case 'children':    return test.children.map(c => testNumber(k, c, node,`${name}-${k}`)).reduce(accumulate, 0);

 405            default: log.warn(`testReference found unknown key '${k}' for path '${name}'`);

 406        }

 407        return -1;

 408    }).reduce(accumulate, 0);

 409}
