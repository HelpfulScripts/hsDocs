  47export class DocsType implements DocsReferenceIdType {

  48    static makeType(mdlType:DocsGenericType, node:DocsNode) {

  49        switch (mdlType.type) {

  50            case 'intrinsic': return new DocsIntrinsicType(mdlType, node);

  51            case 'array': return new DocsArrayType(mdlType, node);

  52            case 'reflection': return new DocsReflectionType(mdlType, node);

  53            case 'stringLiteral': return new DocsStringLiteralType(mdlType, node);

  54            case 'reference': return new DocsReferenceType(mdlType, node);

  55            case 'tuple': return new DocsTupleType(mdlType, node);

  56            case 'union': return new DocsUnionType(mdlType, node);

  57            case 'typeParameter': return new DocsTypeParameterType(mdlType, node);

  58            case 'typeOperator': return new DocsTypeOperatorType(mdlType, node);

  59            case 'query': return new DocsQueryType(mdlType, node);

  60            case 'indexedAccess': return new DocsIndexedAccessType(mdlType, node);

  61            case 'predicate': return new DocsPredicateType(mdlType, node);

  62            case 'unknown': return new DocsUnknownType(mdlType, node);

  63            default: log.warn(`unknown type '${mdlType.type}' for ${node.fullPath} in makeType`);

  64        }

  65    }

  66    type: string;

  67    name: string;

  68    node: DocsNode;

  69    id:   number;

  70    declaration: DocsNode;

  71    constructor(mdlType:DocsGenericType, node:DocsNode) {

  72        this.type = mdlType.type;

  73        this.id   = mdlType.id;

  74        this.name = mdlType.name;

  75        this.node = node;

  76    }

  77

  78    mType() { return m('span.hsdocs_type_unknown', 'default type'); }

  79}

  81class DocsIntrinsicType extends DocsType {

  82    name: string;

  83    constructor(mdlType:DocsGenericType, node:DocsNode) {

  84        super(mdlType, node);

  85        this.name = mdlType.name;

  86    }

  87    mType() {

  88        return m('span.hsdocs_type_intrinsic', this.id? libLinkByPath(this.node.lib, ''+this.id, this.name) : this.name); 

  89    }

  92class DocsReferenceType extends DocsType {

  93    typeArguments: DocsType[];

  94    constructor(mdlType:DocsGenericType, node:DocsNode) {

  95        super(mdlType, node);

  96        this.typeArguments = !mdlType.typeArguments? [] : mdlType.typeArguments.map((t:DocsNamedType) => DocsType.makeType(t, node));

  97    }

  98

  99    mType() {

 100        let refRes = [this.id? libLinkByPath(this.node.lib, ''+this.id, this.name) : this.name];

 101        if (this.typeArguments.length) { 

 102            refRes.push('<');

 103            refRes.push(...this.typeArguments.map(t => t.mType()));

 104            refRes.push('>'); 

 105        }

 106        return m('span.hsdocs_type_reference', refRes);        

 107    }

 108}

 141class DocsTypeParameterType extends DocsType {

 142    constraint: DocsNamedType|DocsTypeOperatorType;

 143    constructor(mdlType:DocsGenericType, node:DocsNode) {

 144        super(mdlType, node);

 145        this.constraint = mdlType.constraint;

 146    }

 147    mType() {

 148        // examples:

 149        // constrained:   http://127.0.0.1/hsDocs/#!/api/hsWidget/hsWidget.Button.Button.attrs

 150        // unconstrained: http://127.0.0.1/hsDocs/#!/api/hsUtil/1

 151        return this.name? m('span.hsdocs_type_typeparameter', this.name) :

 152               this.constraint?.name? m('span.hsdocs_type_typeparameter', `${this.constraint.name}`) :

 153               m('span.hsdocs_type_typeparameter', `${this.constraint.operator} ${this.constraint.target.name}`);

 154    }

 170class DocsReflectionType extends DocsType {

 171    constructor(mdlType:DocsGenericType, node:DocsNode) {

 172        super(mdlType, node);

 173        if (mdlType.declaration) {

 174            const decl:DocsNode = this.declaration = DocsNode.traverse(mdlType.declaration, node.lib, node.fullPath);

 175        }

 176    }

 177

 178    mType() {

 179        const dec = this.declaration;

 180        const rflRes = !dec? 'UNKNOWN' : 

 181            dec.children? [

 182                '{ ', 

 183                ...dec.children

 184                    .map((c:any, i:number) =>

 185                        m('span.named_param',[i>0?', ':'', c.name, ':', 

 186                            c.type?c.type.mType() : '??type??',

 187                            c.defaultValue?'='+c.defaultValue : ''])

 188                    ),

 189                ' }']

 190            : dec.getSignatures()? [

 191                ...dec.getSignatures().map(s => titleArr(s))

 192            ] : dec.kindString

 193        ;

 194        return m('span.hsdocs_type_reflection', rflRes);

 195    }

 198class DocsTupleType extends DocsType {

 199    elements = <DocsType[]>[];

 200    constructor(mdlType:DocsGenericType, node:DocsNode) {

 201        super(mdlType, node);

 202        if (mdlType.elements) {

 203            mdlType.elements.forEach((e:DocsType) => this.elements.push(DocsType.makeType(e, node)));

 204        }

 205    }

 206    mType() {

 207        return m('span.hsdocs_type_tuple', [

 208            '[ ',

 209            ...this.elements.map((e:any, i:number) => [i>0?', ':undefined, e.mType()]),

 210            ' ]'

 211        ]);

 212    }

 215class DocsUnionType extends DocsType {

 223        return m('span.hsdocs_type_union', [...this.types.map((t:any, i:number) => m('span', [i>0?' | ':'', t.mType()]))]);

 224    }
