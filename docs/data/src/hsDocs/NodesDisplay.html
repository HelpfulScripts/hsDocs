  41export function libLinkByPath(lib:string, path:string, name:string, css=''):Vnode {

  53export class DocsComment {

  54    shortText?: string;

  55    returns?: string;

  56    text?: string;

  57    tags?: Tag[];

  58

  59    constructor(comment:json, node:DocsNode) {

  60        this.shortText = comment.shortText;

  61        this.returns = comment.returns;

  62        this.text = comment.text;

  63        this.tags = comment.tags || [];

  64    }

  65

  66    /**

  67     * Report the item's description. This can come in different forms that are all handled here:

  68     * - comment.shortText: <description>

  69     * - comment.text: <description>

  70     * - comment.tags[{tag:'description}, text:<description>]

  71     * Any resulting comment will be translated from markdown to html and returned as a `Vnode`.

  72     * @param short boolean; if true, only the first paragraph of the description will be returned

  73     */

  74    getDescription(short: boolean) {

  75        let text = (this.shortText || '');

  76        if (this.text) { text += '\n'+ (this.text || ''); }

  77        this.tags.map((tag:Tag) => {

  78            switch (tag.tag) {

  79                case 'shortText':

  80                case 'description': text = `${text}\n${tag.text}`; break;

  81                case 'param': text = `${text}\n- **${tag.param}**: ${tag.text}`; break;

  82            }

  83        });

  84        // search for pattern <example...<file...</example>

  85        text = short? '' : text.replace(/<example[^<]*?(<file[\S\s]*?)<\/example>/gi, example);

  86        return m('span.hsdocs_comment_desc', prettifyCode(text, short));

  87    }

  88    

  89    /**

  90     * creates the `returns` message for a function or method.

  91     */

  92    getParams(params?:DocsParameter[]) {

  93        return !params? undefined: m('.hsdocs_comment_params', [

  94            m('.hsdocs_params_title', 'parameters:'),

  95            ...params.map(p => 

  96                m('.hsdocs_comment_param',[

  97                    m('span.hsdocs_param_name', [p.getName(), ': ']), 

  98                    m('span.hsdocs_param_comment', commentLong(p))

  99                ]))

 100            ]);

 101    }

 102    

 103    /**

 104     * creates the `returns` message for a function or method.

 105     */

 106    getReturns() {

 107        let text = this.returns;

 108        return !text? undefined: m('span.hsdocs_comment_return', [            

 109            m('span.hsdocs_comment_return_tag', 'returns:'), 

 110            m('span.hsdocs_comment_return_text', text)

 111        ]);

 112    }

 113

 114    getCommentRemainder() {

 115        const tags = Object.getOwnPropertyNames(this);

 116        return tags.length===0? undefined : m('', tags.map((tag:string) => {

 117            switch(tag) {

 118                case 'tags':        // already handled

 119                case 'shortText':   // already handled

 120                case 'text':        // already handled

 121                case 'description': // already handled

 122                case 'returns':     // already handled

 123                        return undefined;

 124                default: return m('.hsdocs_comment_remainder', [

 125                    m('span.hsdocs_comment_return_tag', tag), 

 126                    m('span.hsdocs_comment_return_text', this[tag])

 127                ]);

 128            }

 129        }));

 130    }

 210export function mSignature(node:DocsNode):Vnode {

 211    const optional = (flags:any) => (flags && flags.isOptional)? '.hsdocs_flag_optional' : '';

 212    const params = (s:DocsParamaterized):Vnode[] => 

 213        !s.parameters? undefined : s.parameters.map((p:DocsParameter, i:number) => { 

 214            const name = p.getName() === '__namedParameters'? '' : p.getName();

 215            return m('span', [

 216                m('span.hsdocs_colon', i>0?', ' : ''),

 217                m('span.hsdocs_signature_param', [

 218                    m(`span.hsdocs_itemname${optional(p.flags)}`, name),

 219                    m('span.hsdocs_colon', name.length>0?':' : ''),     // possibly unnamed parameter

 220                    mType(p)

 221                ])

 222            ]);

 223        });

 224    const sigText = (s:DocsParamaterized) => {

 225        let result:Vnode[];

 226        if (s.type && s.type.declaration) {

 227            const dec = s.type.declaration;

 228            const sig = dec.indexSignature || dec.getSignatures();

 229            if (sig) {

 230                result = [m('span', sig.map((isig:DocsSignature) => params(isig)))];

 231            }

 232        }

 233        result = params(s);

 234        if (result) {

 235            result.unshift(m('span.hsdocs_itemname', ' ('));

 236            result.push(m('span.hsdocs_itemname', ')'));

 237        }

 238        if (s.typeParameter) {

 239            if (!result) { result = [];}

 240            result.unshift(...s.typeParameter.map(t => {

 241                const type = (t.type)? ` extends ${t.type.name}` : '';

 242                return m('span',` <${t.name}${type}>`);

 243            }))

 244        }

 245        return result;

 246    };

 247

 248    const sigs:DocsParamaterized[] = node.getSignatures();

 249    return m('span.hsdocs_signatures', !sigs? sigText(node) : sigs.map(s =>  m('span.hsdocs_signature', sigText(s))));

 250}

 268    }

 387interface TestFunction {

 388    (n:DocsNode): boolean;

 389}

 390interface Access {

 391    text: {true:string; false:string;}[];

 392    sort: TestFunction[];

 393    css: string[];

 394}

 395export function members(node:DocsNode):Vnode {

 396    const not        = (fn:TestFunction) => (n:DocsNode) => !fn(n);

 397    const isStatic   = (n:DocsNode) => n.flags.isStatic === true;

 398    const isExported = (n:DocsNode) => n.flags.isExported === true;

 399    const isPublic   = (n:DocsNode) => (n.flags.isPublic || (!n.flags.isPrivate && !n.flags.isProtected))? true : false;

 400    const filterChildren = (cs:[string,DocsNode[]][], filters:((n:DocsNode)=>boolean)[]):[string,DocsNode[]][] => {

 401        if (filters.length===0) { return cs; }

 402        const remaining = filters.slice(1);

 403        return [

 404            ...filterChildren(cs.map(m => [m[0], m[1].filter(filters[0])]), remaining),

 405            ...filterChildren(cs.map(m => [m[0], m[1].filter(not(filters[0]))]), remaining)

 406        ].filter(m => m[1].length);

 407    };

 408

 409    /** flags to sort by */

 410    let sorts:((n:DocsNode)=>boolean)[];

 411    /** the  */

 412    const access:Access = (node.kindString==='External module')? {

 413            text: [{true:'Static', false: ''}, {true:'Exported', false:'Internal'}],

 414            css: ['.hsdocs_flag_static', '.hsdocs_flag_public'],

 415            sort:   [isStatic, isExported]

 416        } : {

 417            text: [{true:'Static', false: ''}, {true:'Public', false:'Protected or Private'}],

 418            sort: [isStatic, isPublic],

 419            css: ['.hsdocs_flag_static', '.hsdocs_flag_public'],

 420        };

 421    

 422    const grp = node.groups;

 423    if (grp) {

 424        const children:[string,DocsNode[]][] = grp.map(g => [g.title, g.children.map(id => DocSets.getNode(id, node.lib))]);

 425        // flags to sort by: 

 426        const _members = filterChildren(children, access.sort);

 427        return m('.hsdocs_members', _members.map(m => member(m[1], m[0], access)));

 428

 429    } else if (node.parameters) {

 430        return m('.hsdocs_members', node.parameters);

 443function member(nodes:DocsNode[], title:string, access:Access): m.Child {

 444    const directChildren    = ((node:DocsNode) => !node.inheritedFrom);

 445    const inheritedChildren = ((node:DocsNode) =>  node.inheritedFrom);

 446    const [css, flagText]   = nodes.length? getFlagText(nodes[0], access) : ['',''];

 447

 448    const content:Vnode[] = nodes.filter(directChildren)

 449        .map((c:DocsNode) => m('.hsdocs_node', {id:c.fullPath}, itemChild(c)));

 450    const inherited:Vnode[] = nodes.filter(inheritedChildren)

 451        .map((c:DocsNode) => m('.hsdocs_node .hsdocs_node_inherited', {id:c.fullPath}, itemChild(c)));

 452    if (inherited.length>0) {

 453        inherited.unshift(m('div.hsdocs_member_title', `${flagText} Inherited ${title}`));

 454    }

 455    if (content.length>0) {

 456        content.unshift(m('div.hsdocs_member_title', `${flagText} ${title}`));

 457    }

 458    const members:Vnode[] = content.concat(inherited);

 459    return !members.length? '' : m(`.hsdocs_member ${css}`, members);

 460}

 462function itemChild(node:DocsNode): m.Children {

 463    if (node.getSignatures()) {

 464        return node.getSignatures().map((s:DocsNode) => 

 465            m('.hsdocs_child.hsdocs_signature', [titleArr(s, node), m('.hsdocs_sig_comment', s.commentLong())])

 466        );

 467    } else if (node.commentLong()) {

 468        return m('.hsdocs_child', [titleArr(node), m('.hsdocs_sig_comment',node.commentLong())]);

 469    } else if (node.inheritedFrom) {

 470        return m('.hsdocs_child', node.inheritedFrom.name);

 471    } else {

 472        return m('.hsdocs_child', titleArr(node));

 473    }

 485/**

 486 * finds segments of `<code>...</code>` in `comment` and replaces them with a prettified version.

 487 * Currently the function performs operations in this sequence:

 488 * - add indentation for brackets {...}

 489 * - wrap the <code>...</code> part within <pre>...</pre> brackets

 490 * - apply markdown syntax

 491 * - wrap results in a trusted mithril node

 492 * @param comment the comment comment 

 493 * @param short only return the first paragraph

 494 */

 495function prettifyCode(comment:string, short:boolean):Vnode { 

 496    let result = comment;

 497

 498    function braceIndenting(text:string): string {

 499        let indent = 0;

 500        const result = text

 501            .substring(6, text.length-7)    // remove <code> and </code>

 502            .trim()

 503            .replace(/(<)/g, '<').replace(/(>)/g, '>')

 504            .split('\n')

 505            .map((l:string) => {

 506                let oldIndent = indent;

 507                let k = l.trim();

 508                indent += Math.max(-1, Math.min(1, k.split('{').length - k.split('}').length)); 

 509                indent += Math.max(-1, Math.min(1, k.split('[').length - k.split(']').length)); 

 510                return '<span class="hs_code_indent"></span>'.repeat(((indent < oldIndent)?indent:oldIndent)) + k;

 511            })

 512            .join('\n')

 513            .trim();

 514        return '<pre><code>' + result + '</code></pre>';

 515    }

 516

 517    result = result.replace(/<code>([\S\s]*?)<\/code>/gi, braceIndenting);

 518    return m.trust(markDown(result, short, m.route.get()));

 519}
