  48export class DocsComment {

  50    returns?: string;

  51    text?: string;

  52    tags?: Tag[];

  54    constructor(comment:json, node:DocsNode) {

  55        this.shortText = comment.shortText;

  56        this.returns = comment.returns;

  57        this.text = comment.text;

  58        this.tags = comment.tags || [];

  59    }

  60

  61    /**

  62     * Report the item's description. This can come in different forms that are all handled here:

  63     * - comment.shortText: 

  64     * - comment.text: 

  65     * - comment.tags[{tag:'description}, text:]

  66     * Any resulting comment will be translated from markdown to html and returned as a `Vnode`.

  67     * @param short boolean; if true, only the first paragraph of the description will be returned

  68     */

  69    getDescription(short: boolean) {

  70        let text = (this.shortText || '');

  71        if (this.text) { text += '\n'+ (this.text || ''); }

  72        this.tags.map((tag:Tag) => {

  73            switch (tag.tag) {

  74                case 'shortText':

  75                case 'description': text = `${text}\n${tag.text}`; break;

  76                case 'param': text = `${text}\n- **${tag.param}**: ${tag.text}`; break;

  77            }

  78        });

  79        // search for pattern 

  80        text = short? '' : text.replace(//gi, example);

  81        return m('span.hsdocs_comment_desc', prettifyCode(text, short));

  82    }

  83    

  84    /**

  85     * creates the `returns` message for a function or method.

  86     */

  87    getParams(params?:DocsParameter[]) {

  88        return !params? undefined: m('.hsdocs_comment_params', [

  89            m('.hsdocs_params_title', 'parameters:'),

  90            ...params.map(p => 

  91                m('.hsdocs_comment_param',[

  92                    m('span.hsdocs_param_name', [p.getName(), ': ']), 

  93                    m('span.hsdocs_param_comment', commentLong(p))

  94                ]))

  95            ]);

  96    }

  97    

  98    /**

  99     * creates the `returns` message for a function or method.

 100     */

 101    getReturns() {

 102        let text = this.returns;

 103        return !text? undefined: m('span.hsdocs_comment_return', [            

 104            m('span.hsdocs_comment_return_tag', 'returns:'), 

 105            m('span.hsdocs_comment_return_text', text)

 106        ]);

 107    }

 108

 109    getCommentRemainder() {

 110        const tags = Object.getOwnPropertyNames(this);

 111        return tags.length===0? undefined : m('', tags.map((tag:string) => {

 112            switch(tag) {

 113                case 'tags':        // already handled

 114                case 'shortText':   // already handled

 115                case 'text':        // already handled

 116                case 'description': // already handled

 117                case 'returns':     // already handled

 118                        return undefined;

 119                default: return m('.hsdocs_comment_remainder', [

 120                    m('span.hsdocs_comment_return_tag', tag), 

 121                    m('span.hsdocs_comment_return_text', this[tag])

 122                ]);

 123            }

 124        }));

 125    }

 202function mSignature(node:DocsNode):Vnode {

 203    const optional = (flags:any) => (flags && flags.isOptional)? '.hsdocs_flag_optional' : '';

 204    const params = (s:DocsParamaterized):Vnode[] => 

 205        !s.parameters? undefined : s.parameters.map((p:DocsParameter, i:number) => { 

 206            const name = p.getName() === '__namedParameters'? '' : p.getName();

 207            return m('span', [

 208                m('span.hsdocs_colon', i>0?', ' : ''),

 209                m('span.hsdocs_signature_param', [

 210                    m(`span.hsdocs_itemname${optional(p.flags)}`, name),

 211                    m('span.hsdocs_colon', name.length>0?':' : ''),     // possibly unnamed parameter

 212                    mType(p)

 213                ])

 214            ]);

 215        });

 216    const sigText = (s:DocsParamaterized) => {

 217        let result:Vnode[];

 218        if (s.type && s.type.declaration) {

 219            const dec = s.type.declaration;

 220            const sig = dec.indexSignature || dec.getSignatures();

 221            if (sig) {

 222                result = sig.map((isig:DocsSignature) => params(isig));

 223            }

 224        }

 225        result = params(s);

 226        if (result) {

 227            result.unshift(m('span.hsdocs_itemname', ' ('));

 228            result.push(m('span.hsdocs_itemname', ')'));

 229        }

 230        if (s.typeParameter) {

 231            result.unshift(...s.typeParameter.map(t => m('span',` <${t.name}>`)));

 232        }

 233        return result;

 234    };

 235

 236    const sigs:DocsParamaterized[] = node.getSignatures();

 237    return m('span.hsdocs_signatures', !sigs? sigText(node) : sigs.map(s =>  m('span.hsdocs_signature', sigText(s))));

 238}

 348/**

 349 * Main comment processing. The result appears directly below the title in the main panel.

 350 * Function parameters are not reported in short form here since it is assumed they will be listed 

 351 * individually below the main comment

 352 * @param node the module to scan for comments

 353 * @return a vnode representing the comment entries

 354 */

 355export function commentLong(node:DocsNode):Vnode[] {

 356    let content:any[] = [];

 357    if (node.comment) {

 358        content.push(node.comment.getDescription(false));

 359        content.push(node.comment.getParams(node.parameters));

 360        content.push(node.comment.getReturns());

 361        content.push(node.comment.getCommentRemainder());

 362    } else if (node.getSignatures()) {

 363        return node.getSignatures().map(s => commentLong(s));

 364    } else if (node.type) {

 365        return node.type.mType();

 366    }

 367    return content.length? content : undefined;    

 368}

 370interface TestFunction {

 371    (n:DocsNode): boolean;

 372}

 373interface Access {

 374    text: {true:string; false:string;}[];

 375    sort: TestFunction[];

 376    css: string[];

 378export function members(node:DocsNode):Vnode {

 379    const not        = (fn:TestFunction) => (n:DocsNode) => !fn(n);

 380    const isStatic   = (n:DocsNode) => n.flags.isStatic === true;

 381    const isExported = (n:DocsNode) => n.flags.isExported === true;

 382    const isPublic   = (n:DocsNode) => (n.flags.isPublic || (!n.flags.isPrivate && !n.flags.isProtected))? true : false;

 383    const filterChildren = (cs:[string,DocsNode[]][], filters:((n:DocsNode)=>boolean)[]):[string,DocsNode[]][] => {

 384        if (filters.length===0) { return cs; }

 385        const remaining = filters.slice(1);

 386        return [

 387            ...filterChildren(cs.map(m => [m[0], m[1].filter(filters[0])]), remaining),

 388            ...filterChildren(cs.map(m => [m[0], m[1].filter(not(filters[0]))]), remaining)

 389        ].filter(m => m[1].length);

 390    };

 391

 392    /** flags to sort by */

 393    let sorts:((n:DocsNode)=>boolean)[];

 394    /** the  */

 395    const access:Access = (node.kindString==='External module')? {

 396            text: [{true:'Static', false: ''}, {true:'Exported', false:'Internal'}],

 397            css: ['.hsdocs_flag_static', '.hsdocs_flag_public'],

 398            sort:   [isStatic, isExported]

 399        } : {

 400            text: [{true:'Static', false: ''}, {true:'Public', false:'Protected or Private'}],

 401            sort: [isStatic, isPublic],

 402            css: ['.hsdocs_flag_static', '.hsdocs_flag_public'],

 403        };

 404    

 405    const grp = node.groups;

 406    if (grp) {

 407        const children:[string,DocsNode[]][] = grp.map(g => [g.title, g.children.map(id => DocSets.getNode(id, node.lib))]);

 408        // flags to sort by: 

 409        const _members = filterChildren(children, access.sort);

 410        return m('.hsdocs_members', _members.map(m => member(m[1], m[0], access)));

 411

 412    } else if (node.parameters) {

 413        return m('.hsdocs_members', node.parameters);

 414

 415    // } else if (node.getSignatures()) {

 416    //     return m('.hsdocs_members', node.getSignatures().map((s:DocsParamaterized) => s.parameters));

 417    

 418    } else if (node.type && node.type.declaration) {

 419        node.type.declaration.lib = node.lib;

 420        return m('.hsdocs_members', node.type.declaration.members());

 421

 422    }

 423    return m('.hsdocs_members');

 424}

 426function member(nodes:DocsNode[], title:string, access:Access): Vnode {

 427    const directChildren    = ((node:DocsNode) => !node.inheritedFrom);

 428    const inheritedChildren = ((node:DocsNode) =>  node.inheritedFrom);

 429    const [css, flagText]   = nodes.length? getFlagText(nodes[0], access) : ['',''];

 430

 431    const content:Vnode[] = nodes.filter(directChildren)

 432        .map((c:DocsNode) => m('.hsdocs_node', {id:c.fullPath}, itemChild(c)));

 433    const inherited:Vnode[] = nodes.filter(inheritedChildren)

 434        .map((c:DocsNode) => m('.hsdocs_node .hsdocs_node_inherited', {id:c.fullPath}, itemChild(c)));

 435    if (inherited.length>0) {

 436        inherited.unshift(m('div.hsdocs_member_title', `${flagText} Inherited ${title}`));

 437    }

 438    if (content.length>0) {

 439        content.unshift(m('div.hsdocs_member_title', `${flagText} ${title}`));

 440    }

 441    const members:DocsNode[] = content.concat(inherited);

 442    return !members.length? '' : m(`.hsdocs_member ${css}`, members);

 443}

 468/**

 469 * finds segments of `...` in `comment` and replaces them with a prettified version.

 470 * Currently the function performs operations in this sequence:

 471 * - add indentation for brackets {...}

 472 * - wrap the <code>...</code> part within <pre>...</pre> brackets

 473 * - apply markdown syntax

 474 * - wrap results in a trusted mithril node

 475 * @param comment the comment comment 

 476 * @param short only return the first paragraph

 477 */

 478function prettifyCode(comment:string, short:boolean):Vnode { 

 479    let result = comment;

 480

 481    function braceIndenting(text:string): string {

 482        let indent = 0;

 483        const result = text

 484            .substring(6, text.length-7)    // remove  and 

 485            .trim()

 486            .replace(/(<)/g, '<').replace(/(>)/g, '>')

 487            .split('\n')

 488            .map((l:string) => {

 489                let oldIndent = indent;

 490                let k = l.trim();

 491                indent += Math.max(-1, Math.min(1, k.split('{').length - k.split('}').length)); 

 492                indent += Math.max(-1, Math.min(1, k.split('[').length - k.split(']').length)); 

 493                return '"hs-code-indent">'.repeat(((indent < oldIndent)?indent:oldIndent)) + k;

 494            })

 495            .join('\n')

 496            .trim();

 497        return '

' + result + '
';

 498    }

 499

 500    result = result.replace(/([\S\s]*?)<\/code>/gi, braceIndenting);

 501    return m.trust(markDown(result, short, m.route.get()));

 502}
